123 dylib(const char* dir_path, const char* lib_name, bool decorations = add_filename_decorations) {
124 if (dir_path == nullptr) {
125 throw std::invalid_argument("The directory path is null");
126 }
127 if (lib_name == nullptr) {
128 throw std::invalid_argument("The library name is null");
129 }
130
131 std::string final_name = lib_name;
132 std::string final_path = dir_path;
133
134 if (decorations) {
135 final_name = filename_components::prefix + final_name + filename_components::suffix;
136 }
137
138 if (!final_path.empty() && final_path.find_last_of('/') != final_path.size() - 1) {
139 final_path += '/';
140 }
141
142 m_handle = open((final_path + final_name).c_str());
143
144 if (m_handle == nullptr) {
145 throw load_error(
146 "Could not load library \"" + final_path + final_name + "\"\n" +
147 get_error_description()
148 );
149 }
150 }

210 native_symbol_type get_symbol(const char* symbol_name) const {
211 if (symbol_name == nullptr) {
212 throw std::invalid_argument("The symbol name to lookup is null");

220 auto* symbol = locate_symbol(m_handle, symbol_name);

223 throw symbol_error(
224 "Could not get symbol \"" + std::string(symbol_name) + "\"\n" +
225 get_error_description()

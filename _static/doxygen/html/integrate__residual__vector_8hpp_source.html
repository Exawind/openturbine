23 void operator()(size_t i_index) const {
24 auto local_residual = Kokkos::Array<double, 6>{};
25 for (auto j_index = 0U; j_index < num_qps; ++j_index) { // QPs
26 const auto weight = qp_weight_(j_index);
27 const auto coeff_c = weight * shape_deriv_(i_index, j_index);
28 const auto coeff_dig = weight * qp_jacobian_(j_index) * shape_interp_(i_index, j_index);
29 for (auto k = 0U; k < 6U; ++k) {
30 local_residual[k] += coeff_c * qp_Fc_(j_index, k) +
31 coeff_dig * (qp_Fd_(j_index, k) + qp_Fi_(j_index, k) -
32 qp_Fe_(j_index, k) - qp_Fg_(j_index, k));
33 }
34 }
35 for (auto k = 0U; k < 6U; ++k) {
36 residual_vector_terms_(i_elem, i_index, k) = local_residual[k] - node_FX_(i_index, k);
37 }

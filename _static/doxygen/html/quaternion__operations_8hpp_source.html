17) {
18 R(0, 0) = q(0) * q(0) + q(1) * q(1) - q(2) * q(2) - q(3) * q(3);

105template <typename QuaternionInput, typename QuaternionOutput>
106KOKKOS_INLINE_FUNCTION void QuaternionInverse(
107 const QuaternionInput& q_in, const QuaternionOutput& q_out
108) {
109 auto length =
110 Kokkos::sqrt(q_in(0) * q_in(0) + q_in(1) * q_in(1) + q_in(2) * q_in(2) + q_in(3) * q_in(3));
111
112 // Inverse of a quaternion is the conjugate divided by the length
113 q_out(0) = q_in(0) / length;
114 for (int i = 1; i < 4; ++i) {
115 q_out(i) = -q_in(i) / length;
116 }
117}

204 const auto angle = std::sqrt(phi[0] * phi[0] + phi[1] * phi[1] + phi[2] * phi[2]);

211 const auto cos_angle = std::cos(angle / 2.);

225KOKKOS_INLINE_FUNCTION
226Kokkos::Array<double, 4> NormalizeQuaternion(const Kokkos::Array<double, 4>& q) {
227 const auto length_squared = q[0] * q[0] + q[1] * q[1] + q[2] * q[2] + q[3] * q[3];
228
229 // If the length is 1, our work is done
230 if (std::abs(length_squared - 1.) < kTolerance) {
231 return q;
232 }
233
234 // If the length of the quaternion is zero, return a default unit quaternion
235 if (std::abs(length_squared) < kTolerance) {
236 return Kokkos::Array<double, 4>{1., 0., 0., 0.};
237 }
238
239 // Normalize the quaternion
240 const auto length = std::sqrt(length_squared);
241 auto normalized_quaternion = Kokkos::Array<double, 4>{};
242 for (auto k = 0U; k < 4U; ++k) {
243 normalized_quaternion[k] = q[k] / length;
244 }
245 return normalized_quaternion;
246}

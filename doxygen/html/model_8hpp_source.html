41[[nodiscard]] inline Solver<DeviceType> CreateSolver(
42 State<DeviceType>& state, Elements<DeviceType>& elements, Constraints<DeviceType>& constraints
43) {
44 assemble_node_freedom_allocation_table(state, elements, constraints);
45 compute_node_freedom_map_table(state);
46 create_element_freedom_table(elements, state);
47 create_constraint_freedom_table(constraints, state);
48
49 return {
50 state.ID,
51 state.active_dofs,
52 state.node_freedom_map_table,
53 elements.NumberOfNodesPerElement(),
54 elements.NodeStateIndices(),
55 constraints.num_dofs,
56 constraints.base_active_dofs,
57 constraints.target_active_dofs,
58 constraints.base_node_freedom_table,
59 constraints.target_node_freedom_table,
60 constraints.row_range,
61 };

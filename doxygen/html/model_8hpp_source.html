176 void SetBeamVelocityAboutPoint(
177 size_t beam_elem_id, const Array_6& velocity, const Array_3& point
178 ) {
179 const auto& beam_elem = this->beam_elements_[beam_elem_id];
180 for (const auto& node_id : beam_elem.node_ids) {
181 // Get node
182 auto& node = this->GetNode(node_id);
183
184 // Get distance from reference point to node
185 const Array_3 r{node.x[0] - point[0], node.x[1] - point[1], node.x[2] - point[2]};
186
187 // Get angular velocity
188 const Array_3 omega{velocity[3], velocity[4], velocity[5]};
189
190 // Calculate velocity contribution from angular velocity
191 const auto omega_cross_r = CrossProduct(omega, r);
192
193 // Set node translation velocity
194 node.v[0] = velocity[0] + omega_cross_r[0];
195 node.v[1] = velocity[1] + omega_cross_r[1];
196 node.v[2] = velocity[2] + omega_cross_r[2];
197
198 // Set node angular velocity
199 node.v[3] = velocity[3];
200 node.v[4] = velocity[4];
201 node.v[5] = velocity[5];
202 }
203 }

205 void SetBeamAccelerationAboutPoint(
206 size_t beam_elem_id, const Array_6& acceleration, const Array_3& omega, const Array_3& point
207 ) {
208 const auto& beam_elem = this->beam_elements_[beam_elem_id];
209 for (const auto& node_id : beam_elem.node_ids) {
210 // Get node
211 auto& node = this->GetNode(node_id);
212
213 // Get distance from reference point to node
214 const Array_3 r{node.x[0] - point[0], node.x[1] - point[1], node.x[2] - point[2]};
215
216 // Get angular acceleration
217 const Array_3 alpha{acceleration[3], acceleration[4], acceleration[5]};
218
219 // Calculate translational acceleration contribution from angular acceleration
220 const auto alpha_cross_r = CrossProduct(alpha, r);
221
222 // Calculate translational acceleration contribution from angular velocity
223 const auto omega_cross_omega_cross_r = CrossProduct(omega, CrossProduct(omega, r));
224
225 // Set node translation acceleration
226 node.vd[0] = acceleration[0] + alpha_cross_r[0] + omega_cross_omega_cross_r[0];
227 node.vd[1] = acceleration[1] + alpha_cross_r[1] + omega_cross_omega_cross_r[1];
228 node.vd[2] = acceleration[2] + alpha_cross_r[2] + omega_cross_omega_cross_r[2];
229
230 // Set node angular acceleration
231 node.vd[3] = alpha[0];
232 node.vd[4] = alpha[1];
233 node.vd[5] = alpha[2];
234 }

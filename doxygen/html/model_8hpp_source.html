32[[nodiscard]] inline Solver<DeviceType> CreateSolver(
33 State<DeviceType>& state, Elements<DeviceType>& elements, Constraints<DeviceType>& constraints
34) {
35 assemble_node_freedom_allocation_table(state, elements, constraints);
36 compute_node_freedom_map_table(state);
37 create_element_freedom_table(elements, state);
38 create_constraint_freedom_table(constraints, state);
39
40 return {
41 state.ID,
42 state.active_dofs,
43 state.node_freedom_map_table,
44 elements.NumberOfNodesPerElement(),
45 elements.NodeStateIndices(),
46 constraints.num_dofs,
47 constraints.base_active_dofs,
48 constraints.target_active_dofs,
49 constraints.base_node_freedom_table,
50 constraints.target_node_freedom_table,

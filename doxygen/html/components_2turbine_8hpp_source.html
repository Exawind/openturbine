140 void PositionNodes(const TurbineInput& input, Model& model) {
141 // Define origin point for rotations
142 const Array_3 origin{0., 0., 0.};
143
144 // Calculate shaft length from tower top to hub and overhang
145 const auto shaft_length = input.tower_axis_to_rotor_apex / cos(input.shaft_tilt_angle);
146
147 // Rotate tower to align with global Z-axis
148 const auto q_x_to_z = RotationVectorToQuaternion({0., -M_PI / 2., 0.});
149 model.RotateBeamAboutPoint(this->tower.beam_element_id, q_x_to_z, origin);
150
151 // Calculate hub position based on tower top and apex offset
152 const auto& tower_top_node = model.GetNode(this->tower.nodes.back().id);
153 const Array_3 apex_position{
154 tower_top_node.x0[0] - input.tower_axis_to_rotor_apex,
155 tower_top_node.x0[1],
156 tower_top_node.x0[2] + input.tower_top_to_rotor_apex,
157 };
158
159 // Calculate angle between blades
160 const double blade_angle_delta = 2. * M_PI / static_cast<double>(this->blades.size());
161
162 // Create hub node at origin
163 this->hub_node = NodeData(
164 model.AddNode().SetPosition({-input.rotor_apex_to_hub, 0., 0., 1., 0., 0., 0.}).Build()
165 );
166
167 // Create azimuth node at shaft base node
168 this->azimuth_node =
169 NodeData(model.AddNode().SetPosition({shaft_length, 0., 0., 1., 0., 0., 0.}).Build());
170
171 // Create shaft base node relative to hub
172 this->shaft_base_node =
173 NodeData(model.AddNode().SetPosition({shaft_length, 0., 0., 1., 0., 0., 0.}).Build());
174
175 // Create yaw bearing node at tower top position
176 this->yaw_bearing_node = NodeData(
177 model.AddNode().SetPosition(tower_top_node.x0).SetOrientation({1., 0., 0., 0.}).Build()
178 );
179
180 // Create vector of rotor node IDs (hub, azimuth, shaft base)
181 std::vector<size_t> rotor_node_ids{
182 this->hub_node.id, this->azimuth_node.id, this->shaft_base_node.id
183 };
184
185 // Loop over blades
186 for (auto i = 0U; i < this->blades.size(); ++i) {
187 // Get node IDs for this blade
188 std::vector<size_t> node_ids;
189 std::transform(
190 this->blades[i].nodes.begin(), this->blades[i].nodes.end(),
191 std::back_inserter(node_ids),
192 [](const NodeData& node) {
193 return node.id;
194 }
195 );
196
197 // Loop through node IDs and rotate them to align with global Z-axis
198 // then translate to hub radius so blade root is at the hub radius
199 for (const auto& node_id : node_ids) {
200 model.GetNode(node_id)
201 .RotateAboutPoint(q_x_to_z, origin)
202 .Translate({0., 0., input.hub_diameter / 2.});
203 }
204
205 // Add blade apex node at the origin and add node ID to node_ids
206 const auto apex_node_id =
207 model.AddNode().SetPosition({0., 0., 0., 1., 0., 0., 0.}).Build();
208 node_ids.push_back(apex_node_id);
209
210 // Add apex node to the model and rotor node IDs
211 this->apex_nodes.emplace_back(apex_node_id);
212
213 // Calculate cone angle rotation quaternion
214 const auto q_cone = RotationVectorToQuaternion({0., -input.cone_angle, 0.});
215
216 // Calculate azimuth angle rotation quaternion
217 const auto q_azimuth =
218 RotationVectorToQuaternion({static_cast<double>(i) * blade_angle_delta, 0., 0.});
219
220 // Rotate blade nodes and apex node to cone angle and then to azimuth angle
221 for (const auto& node_id : node_ids) {
222 model.GetNode(node_id)
223 .RotateAboutPoint(q_cone, origin)
224 .RotateAboutPoint(q_azimuth, origin);
225 }
226
227 // Add blade and apex node IDs to rotor node IDs
228 rotor_node_ids.insert(rotor_node_ids.end(), node_ids.begin(), node_ids.end());
229 }
230
231 // Create shaft rotation quaternion
232 const auto q_shaft_tilt = RotationVectorToQuaternion({0., input.shaft_tilt_angle, 0.});
233
234 // Rotate rotor nodes by shaft tilt angle and translate to apex position
235 for (const auto& node_id : rotor_node_ids) {
236 model.GetNode(node_id).RotateAboutPoint(q_shaft_tilt, origin).Translate(apex_position);
237 }
238 }

243 void AddConstraints(const TurbineInput& input, Model& model) {
244 // Loop through blades
245 for (auto i = 0U; i < this->blades.size(); ++i) {
246 // Get the blade apex node
247 const auto& apex_node = model.GetNode(this->apex_nodes[i].id);
248
249 // Get the blade root node
250 const auto& root_node = model.GetNode(this->blades[i].nodes[0].id);
251
252 // Calculate the pitch axis for the blade
253 const Array_3 pitch_axis{
254 root_node.x0[0] - apex_node.x0[0],
255 root_node.x0[1] - apex_node.x0[1],
256 root_node.x0[2] - apex_node.x0[2],
257 };
258
259 // Create pitch control constraint
260 this->blade_pitch.emplace_back(model.AddRotationControl(
261 {root_node.id, apex_node.id}, pitch_axis, &this->blade_pitch_control[i]
262 ));
263
264 // Add rigid constraint between hub and blade apex
265 this->apex_to_hub.emplace_back(
266 model.AddRigidJointConstraint({this->hub_node.id, apex_node.id})
267 );
268 }
269
270 // Add rigid constraint between hub and azimuth node
271 this->azimuth_to_hub =
272 ConstraintData(model.AddRigidJointConstraint({this->azimuth_node.id, this->hub_node.id})
273 );
274
275 // Shaft axis constraint
276 const Array_3 shaft_axis{-cos(input.shaft_tilt_angle), 0., sin(input.shaft_tilt_angle)};
277 this->shaft_base_to_azimuth = ConstraintData(model.AddRevoluteJointConstraint(
278 {this->shaft_base_node.id, this->azimuth_node.id}, shaft_axis, &torque_control
279 ));
280
281 // Add constraint from yaw bearing to shaft base
282 this->yaw_bearing_to_shaft_base = ConstraintData(
283 model.AddRigidJointConstraint({this->yaw_bearing_node.id, this->shaft_base_node.id})
284 );
285
286 // Add constraint from tower top to yaw bearing
287 this->tower_top_to_yaw_bearing = ConstraintData(model.AddRotationControl(
288 {this->tower.nodes.back().id, this->yaw_bearing_node.id}, {0., 0., 1.},
289 &this->yaw_control
290 ));
291
292 // Add fixed constraint at the tower base
293 this->tower_base = ConstraintData(model.AddFixedBC(this->tower.nodes.front().id));
294 }

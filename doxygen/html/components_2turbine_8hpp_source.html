310 void PositionNodes(const TurbineInput& input, Model& model) {
311 //--------------------------------------------------------------------------
312 // Preliminary calculations
313 //--------------------------------------------------------------------------
314
315 // Define origin point for rotations
316 const Array_3 origin{0., 0., 0.};
317
318 // Calculate angle between the blades
319 const double blade_angle_delta = 2. * M_PI / static_cast<double>(this->blades.size());
320
321 // Calculate rotation quaternion to align a component from x-axis to z-axis (i.e.
322 // rotate about y-axis by -90 degrees)
323 const auto q_x_to_z = RotationVectorToQuaternion({0., -M_PI / 2., 0.});
324
325 // Calculate cone angle rotation quaternion (rotate about y-axis by -cone angle)
326 const auto q_cone = RotationVectorToQuaternion({0., -input.cone_angle, 0.});
327
328 //--------------------------------------------------------------------------
329 // Position tower
330 //--------------------------------------------------------------------------
331
332 // Rotate tower to align with global Z-axis
333 model.RotateBeamAboutPoint(this->tower.beam_element_id, q_x_to_z, origin);
334
335 // Calculate hub position based on tower top and rotor apex offset
336 const auto& tower_top_node = model.GetNode(this->tower.nodes.back().id);
337 const Array_3 apex_position{
338 tower_top_node.x0[0] - input.tower_axis_to_rotor_apex, // horizontal offset
339 tower_top_node.x0[1],
340 tower_top_node.x0[2] + input.tower_top_to_rotor_apex, // vertical offset
341 };
342
343 //--------------------------------------------------------------------------
344 // Create all intermediate nodes and populate node ID collections
345 //--------------------------------------------------------------------------
346
347 CreateIntermediateNodes(input, model);
348
349 //--------------------------------------------------------------------------
350 // Position blades
351 //--------------------------------------------------------------------------
352
353 // Loop over blades
354 for (auto i = 0U; i < this->blades.size(); ++i) {
355 // Get node IDs for this blade only
356 std::vector<size_t> current_blade_node_ids;
357 std::transform(
358 this->blades[i].nodes.begin(), this->blades[i].nodes.end(),
359 std::back_inserter(current_blade_node_ids),
360 [](const NodeData& node) {
361 return node.id;
362 }
363 );
364
365 //----------------------------------------------------
366 // Blade alignment and transformation
367 //----------------------------------------------------
368
369 // Loop through node IDs and rotate them to align with global Z-axis,
370 // then translate to hub radius so blade root is at the hub radius
371 for (const auto& node_id : current_blade_node_ids) {
372 model.GetNode(node_id)
373 .RotateAboutPoint(q_x_to_z, origin) // Rotate to align with Z-axis
374 .Translate({0., 0., input.hub_diameter / 2.}); // Translate to hub radius
375 }
376
377 // Add blade apex node -> current_blade_node_ids vector
378 current_blade_node_ids.push_back(this->apex_nodes[i].id);
379
380 //----------------------------------------------------
381 // Blade cone and azimuth transformations
382 //----------------------------------------------------
383
384 // Calculate azimuth angle rotation quaternion (rotate about x-axis)
385 const auto q_azimuth =
386 RotationVectorToQuaternion({static_cast<double>(i) * blade_angle_delta, 0., 0.});
387
388 // Rotate blade nodes (including apex node) -> cone angle -> azimuth angle
389 for (const auto& node_id : current_blade_node_ids) {
390 model.GetNode(node_id)
391 .RotateAboutPoint(q_cone, origin) // Rotate to cone angle (about y-axis)
392 .RotateAboutPoint(q_azimuth, origin); // Rotate to azimuth angle (about x-axis)
393 }
394 }
395
396 //--------------------------------------------------------------------------
397 // Position rotor i.e. final rotor assembly
398 //--------------------------------------------------------------------------
399
400 // Create shaft rotation quaternion
401 const auto q_shaft_tilt = RotationVectorToQuaternion({0., input.shaft_tilt_angle, 0.});
402
403 // Rotate rotor nodes by shaft tilt angle and translate to apex position
404 for (const auto& node_id : this->rotor_node_ids) {
405 model.GetNode(node_id)
406 .RotateAboutPoint(q_shaft_tilt, origin) // Rotate about shaft tilt
407 .Translate(apex_position); // Translate to apex position
408 }
409 }

425 void CreateIntermediateNodes(const TurbineInput& input, Model& model) {
426 // Calculate shaft length from tower top to hub and overhang (adjusted for shaft tilt)
427 const auto shaft_length = input.tower_axis_to_rotor_apex / cos(input.shaft_tilt_angle);
428
429 // Get tower top node for yaw bearing positioning
430 const auto& tower_top_node = model.GetNode(this->tower.nodes.back().id);
431
432 //--------------------------------------------------------------------------
433 // Create drivetrain nodes
434 //--------------------------------------------------------------------------
435
436 // Create hub node at origin with hub CM offset
437 this->hub_node = NodeData(
438 model.AddNode().SetPosition({-input.rotor_apex_to_hub, 0., 0., 1., 0., 0., 0.}).Build()
439 );
440
441 // Create azimuth node at shaft length from origin
442 this->azimuth_node =
443 NodeData(model.AddNode().SetPosition({shaft_length, 0., 0., 1., 0., 0., 0.}).Build());
444
445 // Create shaft base node at shaft length from origin
446 this->shaft_base_node =
447 NodeData(model.AddNode().SetPosition({shaft_length, 0., 0., 1., 0., 0., 0.}).Build());
448
449 // Create yaw bearing node at tower top position
450 this->yaw_bearing_node = NodeData(
451 model.AddNode().SetPosition(tower_top_node.x0).SetOrientation({1., 0., 0., 0.}).Build()
452 );
453
454 //--------------------------------------------------------------------------
455 // Create blade apex nodes
456 //--------------------------------------------------------------------------
457
458 this->apex_nodes.reserve(this->blades.size());
459
460 // Create one apex node per blade at origin
461 for (auto i = 0U; i < this->blades.size(); ++i) {
462 const auto apex_node_id =
463 model.AddNode().SetPosition({0., 0., 0., 1., 0., 0., 0.}).Build();
464 this->apex_nodes.emplace_back(apex_node_id);
465 }
466
467 //--------------------------------------------------------------------------
468 // Populate node ID collections
469 //--------------------------------------------------------------------------
470
471 //----------------------------------------------------
472 // Tower nodes
473 //----------------------------------------------------
474
475 // Add tower nodes to tower_node_ids vector
476 this->tower_node_ids.reserve(this->tower.nodes.size());
477 std::transform(
478 this->tower.nodes.begin(), this->tower.nodes.end(),
479 std::back_inserter(this->tower_node_ids),
480 [](const auto& node) {
481 return node.id;
482 }
483 );
484
485 //----------------------------------------------------
486 // Drivetrain nodes
487 //----------------------------------------------------
488
489 // Add drivetrain nodes (yaw bearing, shaft base, azimuth, hub) to drivetrain_node_ids vector
490 this->drivetrain_node_ids.reserve(4);
491 this->drivetrain_node_ids = {
492 this->yaw_bearing_node.id, this->shaft_base_node.id, this->azimuth_node.id,
493 this->hub_node.id
494 };
495
496 //----------------------------------------------------
497 // Blade nodes
498 //----------------------------------------------------
499
500 // Populate blade_node_ids with all blade nodes and apex nodes
501 this->blade_node_ids.reserve(
502 this->blades.size() * (this->blades[0].nodes.size() + 1) // +1 for apex node
503 );
504 for (auto i = 0U; i < this->blades.size(); ++i) {
505 // Add blade structural nodes
506 std::transform(
507 this->blades[i].nodes.begin(), this->blades[i].nodes.end(),
508 std::back_inserter(this->blade_node_ids),
509 [](const NodeData& node) {
510 return node.id;
511 }
512 );
513 // Add apex node for this blade
514 this->blade_node_ids.push_back(this->apex_nodes[i].id);
515 }
516
517 //----------------------------------------------------
518 // Rotor nodes
519 //----------------------------------------------------
520
521 // Add rotor nodes (hub, azimuth, shaft base, and all blade nodes) to rotor_node_ids vector
522 this->rotor_node_ids.reserve(this->blade_node_ids.size() + 3);
523 this->rotor_node_ids.insert(
524 this->rotor_node_ids.end(),
525 {this->hub_node.id, this->azimuth_node.id, this->shaft_base_node.id}
526 );
527 this->rotor_node_ids.insert(
528 this->rotor_node_ids.end(), this->blade_node_ids.begin(), this->blade_node_ids.end()
529 );
530
531 //----------------------------------------------------
532 // Nacelle nodes
533 //----------------------------------------------------
534
535 // Add nacelle nodes (drivetrain nodes + blade nodes) to nacelle_node_ids vector
536 this->nacelle_node_ids.reserve(
537 this->drivetrain_node_ids.size() + this->blade_node_ids.size()
538 );
539 this->nacelle_node_ids.insert(
540 this->nacelle_node_ids.end(), this->drivetrain_node_ids.begin(),
541 this->drivetrain_node_ids.end()
542 );
543 this->nacelle_node_ids.insert(
544 this->nacelle_node_ids.end(), this->blade_node_ids.begin(), this->blade_node_ids.end()
545 );
546
547 //----------------------------------------------------
548 // All turbine nodes
549 //----------------------------------------------------
550
551 // Collect all turbine node IDs (tower + nacelle nodes)
552 this->all_turbine_node_ids.reserve(
553 this->tower_node_ids.size() + this->nacelle_node_ids.size()
554 );
555 this->all_turbine_node_ids.insert(
556 this->all_turbine_node_ids.end(), this->tower_node_ids.begin(),
557 this->tower_node_ids.end()
558 );
559 this->all_turbine_node_ids.insert(
560 this->all_turbine_node_ids.end(), this->nacelle_node_ids.begin(),
561 this->nacelle_node_ids.end()
562 );
563 }

663 void AddConstraints(const TurbineInput& input, Model& model) {
664 //--------------------------------------------------------------------------
665 // Blade control constraints
666 //--------------------------------------------------------------------------
667
668 // Loop through blades
669 for (auto i = 0U; i < this->blades.size(); ++i) {
670 // Get the blade apex node
671 const auto& apex_node = model.GetNode(this->apex_nodes[i].id);
672
673 // Get the blade root node
674 const auto& root_node =
675 model.GetNode(this->blades[i].nodes[0].id); // first node of blade
676
677 // Calculate the pitch axis for the blade (from apex to root)
678 const Array_3 pitch_axis{
679 root_node.x0[0] - apex_node.x0[0],
680 root_node.x0[1] - apex_node.x0[1],
681 root_node.x0[2] - apex_node.x0[2],
682 };
683
684 // Create pitch control constraint
685 this->blade_pitch.emplace_back(model.AddRotationControl(
686 {root_node.id, apex_node.id}, pitch_axis, &this->blade_pitch_control[i]
687 ));
688
689 // Add rigid constraint between hub and blade apex
690 this->apex_to_hub.emplace_back(
691 model.AddRigidJointConstraint({this->hub_node.id, apex_node.id})
692 );
693 }
694
695 //--------------------------------------------------------------------------
696 // Drivetrain constraints
697 //--------------------------------------------------------------------------
698
699 // Add rigid constraint between hub and azimuth node
700 this->azimuth_to_hub =
701 ConstraintData(model.AddRigidJointConstraint({this->azimuth_node.id, this->hub_node.id})
702 );
703
704 // Shaft axis constraint - add revolute joint between shaft base and azimuth node
705 const Array_3 shaft_axis{-cos(input.shaft_tilt_angle), 0., sin(input.shaft_tilt_angle)};
706 this->shaft_base_to_azimuth = ConstraintData(model.AddRevoluteJointConstraint(
707 {this->shaft_base_node.id, this->azimuth_node.id}, shaft_axis, &torque_control
708 ));
709
710 // Add rigid constraint from yaw bearing to shaft base
711 this->yaw_bearing_to_shaft_base = ConstraintData(
712 model.AddRigidJointConstraint({this->yaw_bearing_node.id, this->shaft_base_node.id})
713 );
714
715 //--------------------------------------------------------------------------
716 // Nacelle control constraints
717 //--------------------------------------------------------------------------
718
719 // Add constraint from tower top to yaw bearing
720 this->yaw_control = input.nacelle_yaw_angle;
721 this->tower_top_to_yaw_bearing = ConstraintData(model.AddRotationControl(
722 {this->tower.nodes.back().id, this->yaw_bearing_node.id}, {0., 0., 1.},
723 &this->yaw_control
724 ));

771 void SetInitialDisplacements(const TurbineInput& input, Model& model) {
772 //--------------------------------------------------------------------------
773 // Apply initial blade pitch
774 //--------------------------------------------------------------------------
775 for (auto i = 0U; i < this->blades.size(); ++i) {
776 // Get the blade root and apex nodes
777 const auto& root_node = model.GetNode(this->blades[i].nodes.front().id);
778 const auto& apex_node = model.GetNode(this->apex_nodes[i].id);
779
780 // Calculate the pitch axis
781 const Array_3 pitch_axis{
782 root_node.x0[0] - apex_node.x0[0],
783 root_node.x0[1] - apex_node.x0[1],
784 root_node.x0[2] - apex_node.x0[2],
785 };
786
787 // Create rotation vector about the pitch axis
788 const auto pitch_axis_unit = UnitVector(pitch_axis);
789 const Array_3 rotation_vector{
790 input.blade_pitch_angle * pitch_axis_unit[0],
791 input.blade_pitch_angle * pitch_axis_unit[1],
792 input.blade_pitch_angle * pitch_axis_unit[2]
793 };
794
795 // Calculate pitch rotation quaternion
796 const auto q_pitch = RotationVectorToQuaternion(rotation_vector);
797
798 // Use apex node position as rotation center
799 const Array_3 pitch_center{apex_node.x0[0], apex_node.x0[1], apex_node.x0[2]};
800
801 // Apply pitch rotation as displacement to all blade nodes and apex node
802 for (const auto& blade_node : this->blades[i].nodes) {
803 model.GetNode(blade_node.id).RotateDisplacementAboutPoint(q_pitch, pitch_center);
804 }
805 }
806
807 //--------------------------------------------------------------------------
808 // Apply initial nacelle yaw rotation
809 //--------------------------------------------------------------------------
810
811 // Apply initial yaw rotation if non-zero
812 if (std::abs(input.nacelle_yaw_angle) > kZeroTolerance) {
813 // Get tower top node for rotation center
814 const auto& tower_top_node = model.GetNode(this->tower.nodes.back().id);
815
816 // Create yaw rotation quaternion (rotation about tower Z-axis)
817 const auto q_yaw = RotationVectorToQuaternion({0., 0., input.nacelle_yaw_angle});
818
819 // Rotate all nacelle components about tower top position
820 for (const auto& node_id : this->nacelle_node_ids) {
821 model.GetNode(node_id).RotateDisplacementAboutPoint(
822 q_yaw, {tower_top_node.x0[0], tower_top_node.x0[1], tower_top_node.x0[2]}
823 );
824 }
825 }
826
827 //--------------------------------------------------------------------------
828 // Apply tower base displacement
829 //--------------------------------------------------------------------------
830
831 const auto& tower_base_node = model.GetNode(this->tower.nodes.front().id);
832 const Array_3 ref_tower_base_position{
833 tower_base_node.x0[0], tower_base_node.x0[1], tower_base_node.x0[2]
834 };
835
836 // Calculate translation displacement from reference tower base -> input position
837 const Array_3 tower_base_displacement{
838 input.tower_base_position[0] - ref_tower_base_position[0],
839 input.tower_base_position[1] - ref_tower_base_position[1],
840 input.tower_base_position[2] - ref_tower_base_position[2]
841 };
842 // Get tower base orientation at input position
843 const Array_4 tower_base_orientation{
844 input.tower_base_position[3], input.tower_base_position[4], input.tower_base_position[5],
845 input.tower_base_position[6]
846 };
847
848 // Apply tower base displacement if displacement is non-zero or rotation is non-identity
849 if (Norm(tower_base_displacement) > kZeroTolerance ||
850 !IsIdentityQuaternion(tower_base_orientation, kZeroTolerance)) {
851 // Apply displacement to all turbine nodes
852 for (const auto& node_id : this->all_turbine_node_ids) {
853 // first rotate about original tower base position
854 model.GetNode(node_id).RotateDisplacementAboutPoint(
855 tower_base_orientation, ref_tower_base_position
856 );
857 // then translate to new tower base position
858 model.GetNode(node_id).TranslateDisplacement(tower_base_displacement);
859 }
860 }
861
862 //--------------------------------------------------------------------------
863 // Tower base constraint
864 //--------------------------------------------------------------------------
865
866 // Add prescribed BC constraint at the tower base with initial displacements
867 this->tower_base =
868 ConstraintData(model.AddPrescribedBC(tower_base_node.id, tower_base_node.u));
869 }

881 void SetInitialRotorVelocity(const TurbineInput& input, Model& model) {
882 // Calculate shaft axis in current configuration
883 const auto hub_position = model.GetNode(this->hub_node.id).DisplacedPosition();
884 const auto shaft_base_position = model.GetNode(this->shaft_base_node.id).DisplacedPosition();
885 const Array_3 shaft_axis = UnitVector(
886 {hub_position[0] - shaft_base_position[0], hub_position[1] - shaft_base_position[1],
887 hub_position[2] - shaft_base_position[2]}
888 );
889
890 // Collect all rotor node IDs (hub, azimuth, blade nodes, and apex nodes)
891 std::vector<size_t> rotor_velocity_node_ids{this->hub_node.id, this->azimuth_node.id};
892
893 // Add all blade nodes and apex nodes
894 for (size_t i = 0; i < this->blades.size(); ++i) {
895 // Add blade nodes
896 std::transform(
897 this->blades[i].nodes.begin(), this->blades[i].nodes.end(),
898 std::back_inserter(rotor_velocity_node_ids),
899 [](const auto& blade_node) {
900 return blade_node.id;
901 }
902 );
903 // Add apex node
904 rotor_velocity_node_ids.push_back(this->apex_nodes[i].id);
905 }
906
907 // Create rigid body velocity -> transl. vel = 0, angular vel. about shaft axis
908 const Array_6 rigid_body_velocity{
909 0.,
910 0.,
911 0.,
912 input.rotor_speed * shaft_axis[0],
913 input.rotor_speed * shaft_axis[1],
914 input.rotor_speed * shaft_axis[2]
915 };
916
917 // Apply rotational velocity to all rotor nodes about hub node
918 for (const auto& node_id : rotor_velocity_node_ids) {
919 model.GetNode(node_id).SetVelocityAboutPoint(
920 rigid_body_velocity, {hub_position[0], hub_position[1], hub_position[2]}
921 );
922 }
923 }

208 static Array_3 CalculateX0(
209 const Constraint& constraint, const Node& target_node, const Node& base_node
210 ) {
211 Array_3 x0{0., 0., 0.};
212 // Set X0 to the prescribed displacement for fixed and prescribed BCs i.e. constraints
213 // with 1 node
214 if (GetNumberOfNodes(constraint.type) == 1) {
215 x0[0] = target_node.x[0] - constraint.vec[0];
216 x0[1] = target_node.x[1] - constraint.vec[1];
217 x0[2] = target_node.x[2] - constraint.vec[2];
218 return x0;
219 }
220
221 // Default: set X0 to the relative position between nodes
222 x0[0] = target_node.x[0] - base_node.x[0];
223 x0[1] = target_node.x[1] - base_node.x[1];
224 x0[2] = target_node.x[2] - base_node.x[2];
225 return x0;
226 }

229 static Array_3x3 CalculateAxes(const Constraint& constraint, const Array_3& x0) {
230 Array_3x3 rotation_matrix{};
231 if (constraint.type == ConstraintType::kRevoluteJoint) {
232 constexpr Array_3 x = {1., 0., 0.};
233 const Array_3 x_hat =
234 Norm(constraint.vec) != 0. ? UnitVector(constraint.vec) : UnitVector(x0);
235
236 // Create rotation matrix to rotate x to match vector
237 const auto cross_product = CrossProduct(x, x_hat);
238 const auto dot_product = DotProduct(x_hat, x);
239 const auto k = 1. / (1. + dot_product);
240
241 // Set orthogonal unit vectors from the rotation matrix
242 rotation_matrix[0][0] = cross_product[0] * cross_product[0] * k + dot_product;
243 rotation_matrix[0][1] = cross_product[1] * cross_product[0] * k + cross_product[2];
244 rotation_matrix[0][2] = cross_product[2] * cross_product[0] * k - cross_product[1];
245
246 rotation_matrix[1][0] = cross_product[0] * cross_product[1] * k - cross_product[2];
247 rotation_matrix[1][1] = cross_product[1] * cross_product[1] * k + dot_product;
248 rotation_matrix[1][2] = cross_product[2] * cross_product[1] * k + cross_product[0];
249
250 rotation_matrix[2][0] = cross_product[0] * cross_product[2] * k + cross_product[1];
251 rotation_matrix[2][1] = cross_product[1] * cross_product[2] * k - cross_product[0];
252 rotation_matrix[2][2] = cross_product[2] * cross_product[2] * k + dot_product;
253
254 return rotation_matrix;
255 }
256
257 // Set rotation_matrix to the unit vector of the constraint axis for rotation control
258 if (constraint.type == ConstraintType::kRotationControl) {
259 const auto unit_vector = UnitVector(constraint.vec);
260 rotation_matrix[0][0] = unit_vector[0];
261 rotation_matrix[0][1] = unit_vector[1];
262 rotation_matrix[0][2] = unit_vector[2];
263 return rotation_matrix;
264 }
265
266 // If not a revolute/hinge joint, set rotation_matrix to the input vector
267 rotation_matrix[0][0] = constraint.vec[0];
268 rotation_matrix[0][1] = constraint.vec[1];
269 rotation_matrix[0][2] = constraint.vec[2];
270 return rotation_matrix;
271 }

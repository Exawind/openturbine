264 static std::array<double, 3> CalculateX0(
265 const Constraint& constraint, const Node& target_node, const Node& base_node
266 ) {
267 std::array<double, 3> x0{0., 0., 0.};
268 // Set X0 to the prescribed displacement for fixed and prescribed BCs i.e. constraints
269 // with 1 node
270 if (GetNumberOfNodes(constraint.type) == 1) {
271 x0[0] = target_node.x0[0] - constraint.axis_vector[0];
272 x0[1] = target_node.x0[1] - constraint.axis_vector[1];
273 x0[2] = target_node.x0[2] - constraint.axis_vector[2];
274 return x0;
275 }
276
277 // Default: set X0 to the relative position between nodes
278 x0[0] = target_node.x0[0] - base_node.x0[0];
279 x0[1] = target_node.x0[1] - base_node.x0[1];
280 x0[2] = target_node.x0[2] - base_node.x0[2];
281 return x0;
282 }

285 static std::array<std::array<double, 3>, 3> CalculateAxes(
286 const Constraint& constraint, const std::array<double, 3>& x0
287 ) {
288 std::array<std::array<double, 3>, 3> rotation_matrix{};
289 if (constraint.type == ConstraintType::RevoluteJoint) {
290 constexpr std::array<double, 3> x = {1., 0., 0.};
291 const std::array<double, 3> x_hat = math::Norm(constraint.axis_vector) != 0.
292 ? math::UnitVector(constraint.axis_vector)
293 : math::UnitVector(x0);
294
295 // Create rotation matrix to rotate x to match vector
296 const auto cross_product = math::CrossProduct(x, x_hat);
297 const auto dot_product = math::DotProduct(x_hat, x);
298 const auto k = 1. / (1. + dot_product);
299
300 // Set orthogonal unit vectors from the rotation matrix
301 rotation_matrix[0][0] = cross_product[0] * cross_product[0] * k + dot_product;
302 rotation_matrix[0][1] = cross_product[1] * cross_product[0] * k + cross_product[2];
303 rotation_matrix[0][2] = cross_product[2] * cross_product[0] * k - cross_product[1];
304
305 rotation_matrix[1][0] = cross_product[0] * cross_product[1] * k - cross_product[2];
306 rotation_matrix[1][1] = cross_product[1] * cross_product[1] * k + dot_product;
307 rotation_matrix[1][2] = cross_product[2] * cross_product[1] * k + cross_product[0];
308
309 rotation_matrix[2][0] = cross_product[0] * cross_product[2] * k + cross_product[1];
310 rotation_matrix[2][1] = cross_product[1] * cross_product[2] * k - cross_product[0];
311 rotation_matrix[2][2] = cross_product[2] * cross_product[2] * k + dot_product;
312
313 return rotation_matrix;
314 }
315
316 // Set rotation_matrix to the unit vector of the constraint axis for rotation control
317 if (constraint.type == ConstraintType::RotationControl) {
318 const auto unit_vector = math::UnitVector(constraint.axis_vector);
319 rotation_matrix[0][0] = unit_vector[0];
320 rotation_matrix[0][1] = unit_vector[1];
321 rotation_matrix[0][2] = unit_vector[2];
322 return rotation_matrix;
323 }
324
325 // If not a revolute/hinge joint, set rotation_matrix to the input vector
326 rotation_matrix[0][0] = constraint.axis_vector[0];
327 rotation_matrix[0][1] = constraint.axis_vector[1];
328 rotation_matrix[0][2] = constraint.axis_vector[2];
329 return rotation_matrix;
330 }

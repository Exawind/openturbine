186 static Array_3 CalculateX0(
187 const Constraint& constraint, const Node& target_node, const Node& base_node
188 ) {
189 Array_3 x0{0., 0., 0.};
190 // Set X0 to the prescribed displacement for fixed and prescribed BCs i.e. constraints
191 // with 1 node
192 if (GetNumberOfNodes(constraint.type) == 1) {
193 x0[0] = target_node.x[0] - constraint.vec[0];
194 x0[1] = target_node.x[1] - constraint.vec[1];
195 x0[2] = target_node.x[2] - constraint.vec[2];
196 return x0;
197 }
198
199 // Default: set X0 to the relative position between nodes
200 x0[0] = target_node.x[0] - base_node.x[0];
201 x0[1] = target_node.x[1] - base_node.x[1];
202 x0[2] = target_node.x[2] - base_node.x[2];
203 return x0;
204 }

207 static Array_3x3 CalculateAxes(const Constraint& constraint, const Array_3& x0) {
208 Array_3x3 rotation_matrix{};
209 if (constraint.type == ConstraintType::kRevoluteJoint) {
210 constexpr Array_3 x = {1., 0., 0.};
211 const Array_3 x_hat =
212 Norm(constraint.vec) != 0. ? UnitVector(constraint.vec) : UnitVector(x0);
213
214 // Create rotation matrix to rotate x to match vector
215 const auto cross_product = CrossProduct(x, x_hat);
216 const auto dot_product = DotProduct(x_hat, x);
217 const auto k = 1. / (1. + dot_product);
218
219 // Set orthogonal unit vectors from the rotation matrix
220 rotation_matrix[0][0] = cross_product[0] * cross_product[0] * k + dot_product;
221 rotation_matrix[0][1] = cross_product[1] * cross_product[0] * k + cross_product[2];
222 rotation_matrix[0][2] = cross_product[2] * cross_product[0] * k - cross_product[1];
223
224 rotation_matrix[1][0] = cross_product[0] * cross_product[1] * k - cross_product[2];
225 rotation_matrix[1][1] = cross_product[1] * cross_product[1] * k + dot_product;
226 rotation_matrix[1][2] = cross_product[2] * cross_product[1] * k + cross_product[0];
227
228 rotation_matrix[2][0] = cross_product[0] * cross_product[2] * k + cross_product[1];
229 rotation_matrix[2][1] = cross_product[1] * cross_product[2] * k - cross_product[0];
230 rotation_matrix[2][2] = cross_product[2] * cross_product[2] * k + dot_product;
231
232 return rotation_matrix;
233 }
234
235 // Set rotation_matrix to the unit vector of the constraint axis for rotation control
236 if (constraint.type == ConstraintType::kRotationControl) {
237 const auto unit_vector = UnitVector(constraint.vec);
238 rotation_matrix[0][0] = unit_vector[0];
239 rotation_matrix[0][1] = unit_vector[1];
240 rotation_matrix[0][2] = unit_vector[2];
241 return rotation_matrix;
242 }
243
244 // If not a revolute/hinge joint, set rotation_matrix to the input vector
245 rotation_matrix[0][0] = constraint.vec[0];
246 rotation_matrix[0][1] = constraint.vec[1];
247 rotation_matrix[0][2] = constraint.vec[2];
248 return rotation_matrix;
249 }

267 static std::array<double, 3> CalculateX0(
268 const constraints::Constraint& constraint, const Node& target_node, const Node& base_node
269 ) {
270 std::array<double, 3> x0{0., 0., 0.};
271 // Set X0 to the prescribed displacement for fixed and prescribed BCs i.e. constraints
272 // with 1 node
273 if (GetNumberOfNodes(constraint.type) == 1) {
274 x0[0] = target_node.x0[0] - constraint.axis_vector[0];
275 x0[1] = target_node.x0[1] - constraint.axis_vector[1];
276 x0[2] = target_node.x0[2] - constraint.axis_vector[2];
277 return x0;
278 }
279
280 // Default: set X0 to the relative position between nodes
281 x0[0] = target_node.x0[0] - base_node.x0[0];
282 x0[1] = target_node.x0[1] - base_node.x0[1];
283 x0[2] = target_node.x0[2] - base_node.x0[2];
284 return x0;
285 }

288 static std::array<std::array<double, 3>, 3> CalculateAxes(
289 const constraints::Constraint& constraint, const std::array<double, 3>& x0
290 ) {
291 std::array<std::array<double, 3>, 3> rotation_matrix{};
292 if (constraint.type == constraints::ConstraintType::RevoluteJoint) {
293 constexpr std::array<double, 3> x = {1., 0., 0.};
294 const std::array<double, 3> x_hat = math::Norm(constraint.axis_vector) != 0.
295 ? math::UnitVector(constraint.axis_vector)
296 : math::UnitVector(x0);
297
298 // Create rotation matrix to rotate x to match vector
299 const auto cross_product = math::CrossProduct(x, x_hat);
300 const auto dot_product = math::DotProduct(x_hat, x);
301 const auto k = 1. / (1. + dot_product);
302
303 // Set orthogonal unit vectors from the rotation matrix
304 rotation_matrix[0][0] = cross_product[0] * cross_product[0] * k + dot_product;
305 rotation_matrix[0][1] = cross_product[1] * cross_product[0] * k + cross_product[2];
306 rotation_matrix[0][2] = cross_product[2] * cross_product[0] * k - cross_product[1];
307
308 rotation_matrix[1][0] = cross_product[0] * cross_product[1] * k - cross_product[2];
309 rotation_matrix[1][1] = cross_product[1] * cross_product[1] * k + dot_product;
310 rotation_matrix[1][2] = cross_product[2] * cross_product[1] * k + cross_product[0];
311
312 rotation_matrix[2][0] = cross_product[0] * cross_product[2] * k + cross_product[1];
313 rotation_matrix[2][1] = cross_product[1] * cross_product[2] * k - cross_product[0];
314 rotation_matrix[2][2] = cross_product[2] * cross_product[2] * k + dot_product;
315
316 return rotation_matrix;
317 }
318
319 // Set rotation_matrix to the unit vector of the constraint axis for rotation control
320 if (constraint.type == constraints::ConstraintType::RotationControl) {
321 const auto unit_vector = math::UnitVector(constraint.axis_vector);
322 rotation_matrix[0][0] = unit_vector[0];
323 rotation_matrix[0][1] = unit_vector[1];
324 rotation_matrix[0][2] = unit_vector[2];
325 return rotation_matrix;
326 }
327
328 // If not a revolute/hinge joint, set rotation_matrix to the input vector
329 rotation_matrix[0][0] = constraint.axis_vector[0];
330 rotation_matrix[0][1] = constraint.axis_vector[1];
331 rotation_matrix[0][2] = constraint.axis_vector[2];
332 return rotation_matrix;
333 }

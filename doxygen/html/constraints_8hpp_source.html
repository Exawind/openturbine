261 static Array_3 CalculateX0(
262 const Constraint& constraint, const Node& target_node, const Node& base_node
263 ) {
264 Array_3 x0{0., 0., 0.};
265 // Set X0 to the prescribed displacement for fixed and prescribed BCs i.e. constraints
266 // with 1 node
267 if (GetNumberOfNodes(constraint.type) == 1) {
268 x0[0] = target_node.x[0] - constraint.vec[0];
269 x0[1] = target_node.x[1] - constraint.vec[1];
270 x0[2] = target_node.x[2] - constraint.vec[2];
271 return x0;
272 }
273
274 // Default: set X0 to the relative position between nodes
275 x0[0] = target_node.x[0] - base_node.x[0];
276 x0[1] = target_node.x[1] - base_node.x[1];
277 x0[2] = target_node.x[2] - base_node.x[2];
278 return x0;
279 }

282 static Array_3x3 CalculateAxes(const Constraint& constraint, const Array_3& x0) {
283 Array_3x3 rotation_matrix{};
284 if (constraint.type == ConstraintType::kRevoluteJoint) {
285 constexpr Array_3 x = {1., 0., 0.};
286 const Array_3 x_hat =
287 Norm(constraint.vec) != 0. ? UnitVector(constraint.vec) : UnitVector(x0);
288
289 // Create rotation matrix to rotate x to match vector
290 const auto cross_product = CrossProduct(x, x_hat);
291 const auto dot_product = DotProduct(x_hat, x);
292 const auto k = 1. / (1. + dot_product);
293
294 // Set orthogonal unit vectors from the rotation matrix
295 rotation_matrix[0][0] = cross_product[0] * cross_product[0] * k + dot_product;
296 rotation_matrix[0][1] = cross_product[1] * cross_product[0] * k + cross_product[2];
297 rotation_matrix[0][2] = cross_product[2] * cross_product[0] * k - cross_product[1];
298
299 rotation_matrix[1][0] = cross_product[0] * cross_product[1] * k - cross_product[2];
300 rotation_matrix[1][1] = cross_product[1] * cross_product[1] * k + dot_product;
301 rotation_matrix[1][2] = cross_product[2] * cross_product[1] * k + cross_product[0];
302
303 rotation_matrix[2][0] = cross_product[0] * cross_product[2] * k + cross_product[1];
304 rotation_matrix[2][1] = cross_product[1] * cross_product[2] * k - cross_product[0];
305 rotation_matrix[2][2] = cross_product[2] * cross_product[2] * k + dot_product;
306
307 return rotation_matrix;
308 }
309
310 // Set rotation_matrix to the unit vector of the constraint axis for rotation control
311 if (constraint.type == ConstraintType::kRotationControl) {
312 const auto unit_vector = UnitVector(constraint.vec);
313 rotation_matrix[0][0] = unit_vector[0];
314 rotation_matrix[0][1] = unit_vector[1];
315 rotation_matrix[0][2] = unit_vector[2];
316 return rotation_matrix;
317 }
318
319 // If not a revolute/hinge joint, set rotation_matrix to the input vector
320 rotation_matrix[0][0] = constraint.vec[0];
321 rotation_matrix[0][1] = constraint.vec[1];
322 rotation_matrix[0][2] = constraint.vec[2];
323 return rotation_matrix;
324 }

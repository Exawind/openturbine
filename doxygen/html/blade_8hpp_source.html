137 void CreateNodeGeometry(const BladeInput& input) {
138 const auto n_nodes = input.element_order + 1;
139 const auto n_geometry_pts =
140 std::min({input.ref_axis.coordinate_grid.size(), n_nodes, kMaxGeometryPoints});
141
142 if (n_geometry_pts < n_nodes) {
143 // We need to project from n_geometry_pts -> element_order
144 const std::vector<double> kp_xi(MapGeometricLocations(input.ref_axis.coordinate_grid));
145 const auto [phi_kn_geometry, phi_prime_kn_geometry] =
146 ShapeFunctionMatrices(kp_xi, GenerateGLLPoints(n_geometry_pts - 1));
147 const auto geometry_points = PerformLeastSquaresFitting(
148 n_geometry_pts, phi_kn_geometry, input.ref_axis.coordinates
149 );
150 const auto node_coords =
151 ProjectPointsToTargetPolynomial(n_geometry_pts, n_nodes, geometry_points);
152
153 this->node_coordinates.clear();
154 this->node_coordinates.reserve(node_coords.size());
155 std::copy(
156 node_coords.begin(), node_coords.end(), std::back_inserter(this->node_coordinates)
157 );
158 } else {
159 // Fit node coordinates to key points
160 const std::vector<double> kp_xi(MapGeometricLocations(input.ref_axis.coordinate_grid));
161 const auto [phi_kn, phi_prime_kn] = ShapeFunctionMatrices(kp_xi, this->node_xi);
162 this->node_coordinates =
163 PerformLeastSquaresFitting(n_nodes, phi_kn, input.ref_axis.coordinates);
164 }
165
166 // Calculate tangent vectors at each node
167 this->CalcNodeTangents();
168 }

262 void CalcNodeTangents() {
263 const auto n_nodes{this->node_coordinates.size()};
264
265 // Calculate the derivative shape function matrix for the nodes
266 const auto [phi, phi_prime] = ShapeFunctionMatrices(this->node_xi, this->node_xi);
267
268 // Calculate tangent vectors for each node
269 this->node_tangents.resize(n_nodes, {0., 0., 0.});
270 for (auto i = 0U; i < n_nodes; ++i) {
271 for (auto j = 0U; j < 3; ++j) {
272 for (auto k = 0U; k < n_nodes; ++k) {
273 this->node_tangents[i][j] += phi_prime[k][i] * this->node_coordinates[k][j];
274 }
275 }
276 }
277
278 // Normalize tangent vectors
279 std::transform(
280 this->node_tangents.begin(), this->node_tangents.end(), this->node_tangents.begin(),
281 [](std::array<double, 3>& tangent) {
282 const auto norm = Norm(tangent);
283 std::transform(tangent.begin(), tangent.end(), tangent.begin(), [norm](double v) {
284 return v / norm;
285 });
286 return tangent;
287 }
288 );
289 }

296 static std::vector<BeamSection> BuildBeamSections(const BladeInput& input) {
297 // Extraction section stiffness and mass matrices from blade definition
298 std::vector<BeamSection> sections;
299
300 // Add first section after rotating matrices to account for twist
301 auto twist = LinearInterp(
302 input.sections[0].location, input.ref_axis.twist_grid, input.ref_axis.twist
303 );
304 auto q_twist = RotationVectorToQuaternion({twist * M_PI / 180., 0., 0.});
305 sections.emplace_back(
306 input.sections[0].location, RotateMatrix6(input.sections[0].mass_matrix, q_twist),
307 RotateMatrix6(input.sections[0].stiffness_matrix, q_twist)
308 );
309
310 // Loop through remaining section locations
311 for (auto i = 1U; i < input.sections.size(); ++i) {
312 // Add refinement sections if requested
313 for (auto j = 0U; j < input.section_refinement; ++j) {
314 // Calculate interpolation ratio between bounding sections
315 const auto alpha =
316 static_cast<double>(j + 1) / static_cast<double>(input.section_refinement + 1);
317
318 // Interpolate grid location
319 const auto grid_value = (1. - alpha) * input.sections[i - 1].location +
320 alpha * input.sections[i].location;
321
322 // Interpolate mass and stiffness matrices from bounding sections
323 Array_6x6 mass_matrix;
324 Array_6x6 stiffness_matrix;
325 for (auto mi = 0U; mi < 6; ++mi) {
326 for (auto ni = 0U; ni < 6; ++ni) {
327 mass_matrix[mi][ni] =
328 (1. - alpha) * input.sections[i - 1].mass_matrix[mi][ni] +
329 alpha * input.sections[i].mass_matrix[mi][ni];
330 stiffness_matrix[mi][ni] =
331 (1. - alpha) * input.sections[i - 1].stiffness_matrix[mi][ni] +
332 alpha * input.sections[i].stiffness_matrix[mi][ni];
333 }
334 }
335
336 // Calculate twist at current section location via linear interpolation
337 twist = LinearInterp(
338 input.sections[i].location, input.ref_axis.twist_grid, input.ref_axis.twist
339 );
340
341 // Add refinement section
342 q_twist = RotationVectorToQuaternion({twist * M_PI / 180., 0., 0.});
343 sections.emplace_back(
344 grid_value, RotateMatrix6(mass_matrix, q_twist),
345 RotateMatrix6(stiffness_matrix, q_twist)
346 );
347 }
348
349 // Add ending section
350 twist = LinearInterp(
351 input.sections[i].location, input.ref_axis.twist_grid, input.ref_axis.twist
352 );
353 q_twist = RotationVectorToQuaternion({twist * M_PI / 180., 0., 0.});
354 sections.emplace_back(
355 input.sections[i].location, RotateMatrix6(input.sections[i].mass_matrix, q_twist),
356 RotateMatrix6(input.sections[i].stiffness_matrix, q_twist)
357 );
358 }
359
360 return sections;
361 }

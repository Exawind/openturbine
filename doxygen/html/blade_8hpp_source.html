32 Blade(const BladeInput& input, Model& model) : root_node(kInvalidID) {
33 // Number of nodes in blade
34 const auto n_nodes = input.element_order + 1;
35
36 // Generate node locations within element [-1,1]
37 if (input.node_spacing == BladeInput::NodeSpacing::GaussLobattoLegendre) {
38 this->node_xi = GenerateGLLPoints(input.element_order);
39 } else if (input.node_spacing == BladeInput::NodeSpacing::Linear) {
40 this->node_xi.clear();
41 for (auto i = 0U; i <= input.element_order; ++i) {
42 this->node_xi.emplace_back(
43 2. * static_cast<double>(i) / static_cast<double>(input.element_order) - 1.
44 );
45 }
46 } else {
47 throw("invalid node spacing option");
48 }
49
50 // Fit node coordinates to key points
51 const std::vector<double> kp_xi(MapGeometricLocations(input.ref_axis.coordinate_grid));
52 const auto [phi_kn, phi_prime_kn] = ShapeFunctionMatrices(kp_xi, this->node_xi);
53 const auto node_coordinates =
54 PerformLeastSquaresFitting(n_nodes, phi_kn, input.ref_axis.coordinates);
55
56 const auto node_tangents = CalcNodeTangents(node_coordinates);
57
58 // Add nodes to model
59 std::vector<size_t> node_ids;
60 for (auto i = 0U; i < this->node_xi.size(); ++i) {
61 const auto& pos = node_coordinates[i];
62 const auto q_rot = TangentTwistToQuaternion(node_tangents[i], 0.);
63 const auto node_id =
64 model.AddNode()
65 .SetElemLocation((this->node_xi[i] + 1.) / 2.)
66 .SetPosition(pos[0], pos[1], pos[2], q_rot[0], q_rot[1], q_rot[2], q_rot[3])
67 .Build();
68 node_ids.emplace_back(node_id);
69 this->nodes.emplace_back(node_id);
70 }
71
72 const auto sections = BuildBeamSections(input);
73 std::vector<double> section_grid(sections.size());
74 std::transform(
75 sections.begin(), sections.end(), section_grid.begin(),
76 [](const auto& section) {
77 return section.position;
78 }
79 );
80
81 // Calculate trapezoidal quadrature based on section locations
82 const auto trapezoidal_quadrature = CreateTrapezoidalQuadrature(section_grid);
83
84 // Add beam element and get ID
85 this->beam_element_id = model.AddBeamElement(node_ids, sections, trapezoidal_quadrature);

89 //----------------------------------------------------------------------

92 input.root.position[0], input.root.position[1], input.root.position[2]

96 input.root.position[6]

103 model.RotateBeamAboutPoint(this->beam_element_id, root_orientation, root_location);
104 model.SetBeamVelocityAboutPoint(this->beam_element_id, input.root.velocity, root_location);
105 model.SetBeamAccelerationAboutPoint(
106 this->beam_element_id, input.root.acceleration, root_omega, root_location
107 );

110 // Add blade root node and connect to first blade node
111 //----------------------------------------------------------------------

138 [[nodiscard]] std::vector<std::array<double, 3>> CalcNodeTangents(
139 const std::vector<std::array<double, 3>>& node_coordinates
140 ) const {
141 const auto n_nodes{node_coordinates.size()};
142
143 // Calculate the derivative shape function matrix for the nodes
144 const auto [phi, phi_prime] = ShapeFunctionMatrices(this->node_xi, this->node_xi);
145
146 // Calculate tangent vectors for each node
147 std::vector<std::array<double, 3>> node_tangents(n_nodes, {0., 0., 0.});
148 for (auto i = 0U; i < n_nodes; ++i) {
149 for (auto j = 0U; j < 3; ++j) {
150 for (auto k = 0U; k < n_nodes; ++k) {
151 node_tangents[i][j] += phi_prime[k][i] * node_coordinates[k][j];
152 }
153 }
154 }
155 std::transform(
156 node_tangents.begin(), node_tangents.end(), node_tangents.begin(),
157 [](std::array<double, 3>& tangent) {
158 const auto norm = Norm(tangent);
159 std::transform(tangent.begin(), tangent.end(), tangent.begin(), [norm](double v) {
160 return v / norm;
161 });
162 return tangent;
163 }
164 );
165
166 return node_tangents;
167 }

169 static std::vector<BeamSection> BuildBeamSections(const BladeInput& input) {
170 // Extraction section stiffness and mass matrices from blade definition
171 std::vector<BeamSection> sections;
172
173 // Add first section after rotating matrices to account for twist
174 auto twist = LinearInterp(
175 input.sections[0].location, input.ref_axis.twist_grid, input.ref_axis.twist
176 );
177 auto q_twist = RotationVectorToQuaternion({twist * M_PI / 180., 0., 0.});
178 sections.emplace_back(
179 input.sections[0].location, RotateMatrix6(input.sections[0].mass_matrix, q_twist),
180 RotateMatrix6(input.sections[0].stiffness_matrix, q_twist)
181 );
182
183 // Loop through remaining section locations
184 for (auto i = 1U; i < input.sections.size(); ++i) {
185 // Add refinement sections if requested
186 for (auto j = 0U; j < input.section_refinement; ++j) {
187 // Calculate interpolation ratio between bounding sections
188 const auto alpha =
189 static_cast<double>(j + 1) / static_cast<double>(input.section_refinement + 1);
190
191 // Interpolate grid location
192 const auto grid_value = (1. - alpha) * input.sections[i - 1].location +
193 alpha * input.sections[i].location;
194
195 // Interpolate mass and stiffness matrices from bounding sections
196 Array_6x6 mass_matrix;
197 Array_6x6 stiffness_matrix;
198 for (auto mi = 0U; mi < 6; ++mi) {
199 for (auto ni = 0U; ni < 6; ++ni) {
200 mass_matrix[mi][ni] =
201 (1. - alpha) * input.sections[i - 1].mass_matrix[mi][ni] +
202 alpha * input.sections[i].mass_matrix[mi][ni];
203 stiffness_matrix[mi][ni] =
204 (1. - alpha) * input.sections[i - 1].stiffness_matrix[mi][ni] +
205 alpha * input.sections[i].stiffness_matrix[mi][ni];
206 }
207 }
208
209 // Calculate twist at current section location via linear interpolation
210 twist = LinearInterp(
211 input.sections[i].location, input.ref_axis.twist_grid, input.ref_axis.twist
212 );
213
214 // Add refinement section
215 q_twist = RotationVectorToQuaternion({twist * M_PI / 180., 0., 0.});
216 sections.emplace_back(
217 grid_value, RotateMatrix6(mass_matrix, q_twist),
218 RotateMatrix6(stiffness_matrix, q_twist)
219 );
220 }
221
222 // Add ending section
223 twist = LinearInterp(
224 input.sections[i].location, input.ref_axis.twist_grid, input.ref_axis.twist
225 );
226 q_twist = RotationVectorToQuaternion({twist * M_PI / 180., 0., 0.});
227 sections.emplace_back(
228 input.sections[i].location, RotateMatrix6(input.sections[i].mass_matrix, q_twist),
229 RotateMatrix6(input.sections[i].stiffness_matrix, q_twist)
230 );
231 }
232
233 return sections;
234 }

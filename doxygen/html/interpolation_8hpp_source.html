17inline void LinearInterpWeights(
18 double x, const std::vector<double>& xs, std::vector<double>& weights
19) {
20 const auto n = xs.size();
21 weights.assign(n, 0.);
22
23 const auto lower = std::lower_bound(xs.begin(), xs.end(), x);
24 // If x is less than the first node, first weight -> 1 and our work is done
25 if (lower == xs.begin()) {
26 weights.front() = 1.;
27 return;
28 }
29 // If x is greater than the last node, last weight -> 1 and our work is done
30 if (lower == xs.end()) {
31 weights.back() = 1.;
32 return;
33 }
34 // x is between two nodes, so compute weights for closest nodes
35 const auto index = static_cast<unsigned>(std::distance(xs.begin(), lower));
36 const auto lower_loc = xs[index - 1];
37 const auto upper_loc = xs[index];
38 const auto weight_upper = (x - lower_loc) / (upper_loc - lower_loc);
39 weights[index - 1] = 1. - weight_upper;
40 weights[index] = weight_upper;
41}

111 std::vector<std::vector<double>> denom(n, std::vector<double>(n));
112 for (size_t i = 0; i < n; ++i) {
113 for (size_t k = 0; k < n; ++k) {

121 for (size_t j = 0; j < n; ++j) {

124 for (size_t k = 0; k < n; ++k) {

146inline double LegendrePolynomial(const size_t n, const double x) {
147 // Base cases
148 if (n == 0) {
149 return 1.;
150 }
151 if (n == 1) {
152 return x;
153 }
154
155 // Compute the nth Legendre polynomial iteratively
156 double p_n_minus_2{1.}; // P_{n-2}(x)
157 double p_n_minus_1{x}; // P_{n-1}(x)
158 double p_n{0.}; // P_n(x)
159 for (size_t i = 2; i <= n; ++i) {
160 const auto i_double = static_cast<double>(i);
161 p_n = ((2. * i_double - 1.) * x * p_n_minus_1 - (i_double - 1.) * p_n_minus_2) / i_double;
162 p_n_minus_2 = p_n_minus_1;
163 p_n_minus_1 = p_n;
164 }
165 return p_n;
166}

185inline std::vector<double> GenerateGLLPoints(const size_t order) {
186 if (order < 1) {
187 throw std::invalid_argument("Polynomial order must be >= 1");
188 }
189
190 const size_t n_nodes = order + 1;
191 auto gll_points = std::vector<double>(n_nodes, 0.);
192 gll_points.resize(n_nodes);
193
194 // Set the endpoints fixed at [-1, 1]
195 gll_points.front() = -1.;
196 gll_points.back() = 1.;
197
198 // Find interior GLL points (1, ..., order - 1) using Newton-Raphson iteration
199 std::vector<double> legendre_poly(n_nodes, 0.);
200 for (size_t i = 1; i < order; ++i) {
201 // Initial guess using Chebyshev-Gauss-Lobatto nodes
202 auto x_it = -std::cos(static_cast<double>(i) * M_PI / static_cast<double>(order));
203
204 bool converged{false};
205 for (size_t it = 0; it < kMaxIterations; ++it) {
206 const auto x_old = x_it;
207
208 // Compute Legendre polynomials up to order n
209 for (size_t k = 0; k < n_nodes; ++k) {
210 legendre_poly[k] = LegendrePolynomial(k, x_it);
211 }
212
213 // Newton update: x_{n+1} = x_n - f(x_n)/f'(x_n)
214 const auto numerator = x_it * legendre_poly[n_nodes - 1] - legendre_poly[n_nodes - 2];
215 const auto denominator = static_cast<double>(n_nodes) * legendre_poly[n_nodes - 1];
216 x_it -= numerator / denominator;
217
218 // Check for convergence
219 if (std::abs(x_it - x_old) <= kConvergenceTolerance) {
220 converged = true;
221 break;
222 }
223 }
224
225 if (!converged) {
226 throw std::runtime_error(
227 "Newton-Raphson iteration failed to converge for GLL point index " +
228 std::to_string(i)
229 );
230 }
231
232 gll_points[i] = x_it;
233 }
234
235 return gll_points;
236}

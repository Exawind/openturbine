15inline void LinearInterpWeights(
16 double x, const std::vector<double>& xs, std::vector<double>& weights
17) {
18 const auto n = xs.size();
19 weights.assign(n, 0.);
20
21 const auto lower = std::lower_bound(xs.begin(), xs.end(), x);
22 // If x is less than the first node, first weight -> 1 and our work is done
23 if (lower == xs.begin()) {
24 weights.front() = 1.;
25 return;
26 }
27 // If x is greater than the last node, last weight -> 1 and our work is done
28 if (lower == xs.end()) {
29 weights.back() = 1.;
30 return;
31 }
32 // x is between two nodes, so compute weights for closest nodes
33 const auto index = static_cast<unsigned>(std::distance(xs.begin(), lower));
34 const auto lower_loc = xs[index - 1];
35 const auto upper_loc = xs[index];
36 const auto weight_upper = (x - lower_loc) / (upper_loc - lower_loc);
37 weights[index - 1] = 1. - weight_upper;
38 weights[index] = weight_upper;

124inline double LegendrePolynomial(const size_t n, const double x) {
125 // Base cases
126 if (n == 0) {
127 return 1.;
128 }
129 if (n == 1) {
130 return x;
131 }
132
133 // Compute the nth Legendre polynomial iteratively
134 double p_n_minus_2{1.}; // P_{n-2}(x)
135 double p_n_minus_1{x}; // P_{n-1}(x)
136 double p_n{0.}; // P_n(x)
137 for (size_t i = 2; i <= n; ++i) {
138 const auto i_double = static_cast<double>(i);
139 p_n = ((2. * i_double - 1.) * x * p_n_minus_1 - (i_double - 1.) * p_n_minus_2) / i_double;
140 p_n_minus_2 = p_n_minus_1;
141 p_n_minus_1 = p_n;
142 }
143 return p_n;
144}

163inline std::vector<double> GenerateGLLPoints(const size_t order) {
164 if (order < 1) {
165 throw std::invalid_argument("Polynomial order must be >= 1");
166 }
167
168 const size_t n_nodes = order + 1;
169 auto gll_points = std::vector<double>(n_nodes, 0.);
170 gll_points.resize(n_nodes);
171
172 // Set the endpoints fixed at [-1, 1]
173 gll_points.front() = -1.;
174 gll_points.back() = 1.;
175
176 // Find interior GLL points (1, ..., order - 1) using Newton-Raphson iteration
177 std::vector<double> legendre_poly(n_nodes, 0.);
178 for (size_t i = 1; i < order; ++i) {
179 // Initial guess using Chebyshev-Gauss-Lobatto nodes
180 auto x_it = -std::cos(static_cast<double>(i) * M_PI / static_cast<double>(order));
181
182 bool converged{false};
183 for (size_t it = 0; it < kMaxIterations; ++it) {
184 const auto x_old = x_it;
185
186 // Compute Legendre polynomials up to order n
187 for (size_t k = 0; k < n_nodes; ++k) {
188 legendre_poly[k] = LegendrePolynomial(k, x_it);
189 }
190
191 // Newton update: x_{n+1} = x_n - f(x_n)/f'(x_n)
192 const auto numerator = x_it * legendre_poly[n_nodes - 1] - legendre_poly[n_nodes - 2];
193 const auto denominator = static_cast<double>(n_nodes) * legendre_poly[n_nodes - 1];
194 x_it -= numerator / denominator;
195
196 // Check for convergence
197 if (std::abs(x_it - x_old) <= kConvergenceTolerance) {
198 converged = true;
199 break;
200 }
201 }
202
203 if (!converged) {
204 throw std::runtime_error(
205 "Newton-Raphson iteration failed to converge for GLL point index " +
206 std::to_string(i)
207 );
208 }
209
210 gll_points[i] = x_it;
211 }
212
213 return gll_points;

32static HollowCircleProperties CalculateHollowCircleProperties(
33 double outer_diameter, double wall_thickness, double nu = 0.33
34) {
35 //--------------------------------------------------------------------------
36 // Check input arguments
37 //--------------------------------------------------------------------------
38 if (wall_thickness >= outer_diameter / 2.) {
39 throw std::invalid_argument("Wall thickness must be less than outer radius");
40 }
41
42 //--------------------------------------------------------------------------
43 // Calculate geometric properties
44 //--------------------------------------------------------------------------
45 const double outer_radius = outer_diameter / 2.;
46 const double inner_radius = outer_radius - wall_thickness;
47
48 // Calculate geometric properties using difference of circles
49 const double area = M_PI * (std::pow(outer_radius, 2) - std::pow(inner_radius, 2));
50 const double Ixx = M_PI * (std::pow(outer_radius, 4) - std::pow(inner_radius, 4)) / 4.;
51 const double Iyy = Ixx; // Circular symmetry
52 const double J = M_PI * (std::pow(outer_radius, 4) - std::pow(inner_radius, 4)) / 2.;
53
54 // Shear correction factors for hollow circular sections
55 const double kx = (6. * (1. + nu)) / (7. + 6. * nu); // Timoshenko-Ehrenfest beam theory
56 const double ky = kx; // Circular symmetry
57
58 return HollowCircleProperties{area, Ixx, Iyy, J, kx, ky};
59}

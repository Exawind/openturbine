331 void Validate() const {
332 // Check if the number of blades is valid
333 if (n_blades < 1) {
334 throw std::runtime_error("Invalid number of blades. Must be at least 1.");
335 }
336
337 // Validate hub and nacelle data - should have exactly one mesh point each
338 if (hub.NumberOfMeshPoints() != 1 || nacelle.NumberOfMeshPoints() != 1) {
339 throw std::runtime_error("Hub and nacelle should have exactly one mesh point each.");
340 }
341
342 // Check if the number of blade roots matches the number of blades
343 if (blade_roots.NumberOfMeshPoints() != NumberOfBlades()) {
344 throw std::runtime_error("Number of blade roots does not match number of blades.");
345 }
346
347 // Check if the blade nodes to blade number mapping is valid - should be the same size
348 if (blade_nodes_to_blade_num_mapping.size() != blade_nodes.NumberOfMeshPoints()) {
349 throw std::runtime_error("Blade node to blade number mapping size mismatch.");
350 }
351
352 // Check if the node indices by blade are valid - should be same size as number of blades
353 if (node_indices_by_blade.size() != NumberOfBlades()) {
354 throw std::runtime_error("Node indices by blade size mismatch.");
355 }
356
357 // Check if the total number of blade nodes is valid - should be the same as the number of
358 // aggregrated blade nodes
359 size_t total_nodes = 0;
360 for (const auto& bl : node_indices_by_blade) {
361 total_nodes += bl.size();
362 }
363 if (total_nodes != blade_nodes.NumberOfMeshPoints()) {
364 throw std::runtime_error("Total number of blade nodes mismatch.");
365 }

385 void SetBladeNodeMotion(
386 size_t blade_number, size_t node_number, const std::array<double, 7>& position,
387 const std::array<double, 6>& velocity, const std::array<double, 6>& acceleration
388 ) {
389 // Check if the blade and node numbers are within the valid range
390 if (blade_number >= NumberOfBlades() ||
391 node_number >= node_indices_by_blade[blade_number].size()) {
392 throw std::out_of_range("Blade or node number out of range.");
393 }
394
395 // Rotation to convert blade orientation
396 const auto r_x2z = RotationVectorToQuaternion({0., M_PI / 2., 0.});
397
398 // Original orientation
399 const std::array<double, 4> r{position[3], position[4], position[5], position[6]};
400
401 // Converted orientation
402 const auto r_adi = QuaternionCompose(r, r_x2z);
403
404 // Position with new orientation
405 const std::array<double, 7> position_new{position[0], position[1], position[2], r_adi[0],
406 r_adi[1], r_adi[2], r_adi[3]};
407
408 const size_t node_index = node_indices_by_blade[blade_number][node_number];
409 blade_nodes.SetValues(node_index, position_new, velocity, acceleration);
410 }

529 void InitializeBlades(const TurbineConfig& tc) {
530 size_t i_blade{0};
531 size_t i_node_total{0};
532
533 // Initialize blade roots and nodes
534 for (const auto& blade : tc.blade_initial_states) {
535 // Initialize blade root
536 this->SetBladeRootMotion(
537 i_blade, blade.root_initial_position, std::array<double, 6>{0., 0., 0., 0., 0., 0.},
538 std::array<double, 6>{0., 0., 0., 0., 0., 0.}
539 );
540
541 // Initialize blade nodes
542 size_t i_node{0};
543 for (const auto& position : blade.node_initial_positions) {
544 node_indices_by_blade[i_blade].emplace_back(i_node_total);
545 blade_nodes_to_blade_num_mapping.emplace_back(static_cast<int32_t>(i_blade + 1));
546
547 this->SetBladeNodeMotion(
548 i_blade, i_node, position,
549 std::array<double, 6>{0., 0., 0., 0., 0., 0.}, // velocity
550 std::array<double, 6>{0., 0., 0., 0., 0., 0.} // acceleration
551 );
552 ++i_node;
553 ++i_node_total;
554 }
555 ++i_blade;
556 }

745 void SetupRotors(const std::vector<TurbineConfig>& turbine_configs) {
746 auto ADI_C_SetupRotor = lib_.get_function<
747 void(int*, int*, const float*, float*, double*, float*, double*, int*, float*, double*, int*, float*, double*, int*, int*, char*)>(
748 "ADI_C_SetupRotor"
749 );
750
751 // Loop through turbine configurations
752 int32_t turbine_number{0};
753 for (const auto& tc : turbine_configs) {
754 // Turbine number is 1 indexed i.e. 1, 2, 3, ...
755 ++turbine_number;
756
757 // Convert bool -> int32_t to pass to the Fortran routine
758 int32_t is_horizontal_axis_int = tc.is_horizontal_axis ? 1 : 0;
759
760 // Validate the turbine config
761 tc.Validate();
762
763 // Create new turbine data to store the updates
764 // Note: TurbineData and MeshData are validated during construction, so no need to
765 // perform additional checks here
766 turbines.emplace_back(tc);
767 auto& td = turbines.back();
768
769 // Call setup rotor for each turbine
770 ADI_C_SetupRotor(
771 &turbine_number, // input: current turbine number
772 &is_horizontal_axis_int, // input: 1: HAWT, 0: VAWT or cross-flow
773 tc.reference_position.data(), // input: turbine reference position
774 td.hub.position[0].data(), // input: initial hub position
775 td.hub.orientation[0][0].data(), // input: initial hub orientation
776 td.nacelle.position[0].data(), // input: initial nacelle position
777 td.nacelle.orientation[0][0].data(), // input: initial nacelle orientation

321 void Validate() const {
322 // Check if the number of blades is valid
323 if (n_blades < 1) {
324 throw std::runtime_error("Invalid number of blades. Must be at least 1.");
325 }
326
327 // Validate hub and nacelle data - should have exactly one mesh point each
328 if (hub.NumberOfMeshPoints() != 1 || nacelle.NumberOfMeshPoints() != 1) {
329 throw std::runtime_error("Hub and nacelle should have exactly one mesh point each.");
330 }
331
332 // Check if the number of blade roots matches the number of blades
333 if (blade_roots.NumberOfMeshPoints() != NumberOfBlades()) {
334 throw std::runtime_error("Number of blade roots does not match number of blades.");
335 }
336
337 // Check if the blade nodes to blade number mapping is valid - should be the same size
338 if (blade_nodes_to_blade_num_mapping.size() != blade_nodes.NumberOfMeshPoints()) {
339 throw std::runtime_error("Blade node to blade number mapping size mismatch.");
340 }
341
342 // Check if the node indices by blade are valid - should be same size as number of blades
343 if (node_indices_by_blade.size() != NumberOfBlades()) {
344 throw std::runtime_error("Node indices by blade size mismatch.");
345 }
346
347 // Check if the total number of blade nodes is valid - should be the same as the number of
348 // aggregrated blade nodes
349 size_t total_nodes = 0;
350 for (const auto& bl : node_indices_by_blade) {
351 total_nodes += bl.size();

734 "ADI_C_SetupRotor"
735 );
736
737 // Loop through turbine configurations
738 int32_t turbine_number{0};
739 for (const auto& tc : turbine_configs) {
740 // Turbine number is 1 indexed i.e. 1, 2, 3, ...

757 &turbine_number, // input: current turbine number

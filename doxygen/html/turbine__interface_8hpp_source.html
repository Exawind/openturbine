113 [[nodiscard]] bool Step() {
114 // Apply controller if available
115 if (controller) {
116 ApplyController();
117 }
118
119 // Update the host state with current node loads
120 Kokkos::deep_copy(this->host_state.f, 0.);

124 // Solve for state at end of step
125 auto converged = openturbine::Step(

208 [[nodiscard]] double CalculateAzimuthAngle() const {
209 const auto azimuth_constraint_id = this->turbine.shaft_base_to_azimuth.id;

211
212 // Normalize azimuth angle to range [0, 2œÄ) radians
213 azimuth = std::fmod(azimuth, 2. * M_PI);

234 void InitializeController(
235 const components::TurbineInput& turbine_input,
236 const components::SolutionInput& solution_input
237 ) {
238 if (!controller) {

242 // Set controller constant parameters
243 controller->io.dt = solution_input.time_step; // Time step size (seconds)
244 controller->io.pitch_actuator_type_req = 0; // Pitch position actuator
245 controller->io.pitch_control_type = 0; // Collective pitch control
246 controller->io.n_blades = turbine_input.blades.size(); // Number of blades
247
248 // Set controller initial values
249 controller->io.time = 0.; // Current time (seconds)
250 controller->io.azimuth_angle = turbine_input.azimuth_angle; // Initial azimuth
251 controller->io.pitch_blade1_actual = turbine_input.blade_pitch_angle; // Blade pitch (rad)
252 controller->io.pitch_blade2_actual = turbine_input.blade_pitch_angle; // Blade pitch (rad)
253 controller->io.pitch_blade3_actual = turbine_input.blade_pitch_angle; // Blade pitch (rad)
254 controller->io.generator_speed_actual =
255 turbine_input.rotor_speed * turbine_input.gear_box_ratio; // Generator speed (rad/s)
256 controller->io.generator_torque_actual =
257 turbine_input.generator_power /
258 (turbine_input.rotor_speed * turbine_input.gear_box_ratio); // Generator torque
259 controller->io.generator_power_actual =
260 turbine_input.generator_power; // Generator power (W)
261 controller->io.rotor_speed_actual = turbine_input.rotor_speed; // Rotor speed (rad/s)
262 controller->io.horizontal_wind_speed = turbine_input.hub_wind_speed; // Hub wind speed (m/s)
263
264 // Signal first call to controller
265 controller->io.status = 0;

267 // Make first call to controller to initialize
268 controller->CallController();

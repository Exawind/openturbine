34 KOKKOS_FUNCTION
35 void operator()(const int i_elem) const {
36 const auto num_nodes = num_nodes_per_element(i_elem);
37 const auto num_qps = num_qps_per_element(i_elem);
38 const auto shape_deriv = Kokkos::subview(
39 shape_derivative, i_elem, Kokkos::make_pair(size_t{0U}, num_nodes),
40 Kokkos::make_pair(size_t{0U}, num_qps)
41 );
42 const auto qp_pos_deriv = Kokkos::subview(
43 qp_position_derivative, i_elem, Kokkos::make_pair(size_t{0U}, num_qps), Kokkos::ALL
44 );
45 const auto node_pos = Kokkos::subview(
46 node_position_rotation, i_elem, Kokkos::make_pair(size_t{0U}, num_nodes),
47 Kokkos::make_pair(0U, 3U)
48 );
49 const auto qp_jacob =
50 Kokkos::subview(qp_jacobian, i_elem, Kokkos::make_pair(size_t{0U}, num_qps));
51
52 // Interpolate position derivatives at quadrature points using shape functions
53 // qp_pos_deriv = Σ(dN/dξ * node_pos)
54 InterpVector3(shape_deriv, node_pos, qp_pos_deriv);
55
56 for (auto j_qp = 0U; j_qp < num_qps; ++j_qp) {
57 // Calculate Jacobian - this is a scalar that represents the "stretching" factor between
58 // parametric (ξ) and physical (x) space
59 // J = |dx/dξ| = sqrt((dx/dξ)² + (dy/dξ)² + (dz/dξ)²)
60 const auto jacobian = Kokkos::sqrt(
61 Kokkos::pow(qp_pos_deriv(j_qp, 0), 2.) + Kokkos::pow(qp_pos_deriv(j_qp, 1), 2.) +
62 Kokkos::pow(qp_pos_deriv(j_qp, 2), 2.)
63 );
64 qp_jacob(j_qp) = jacobian;
65
66 // Normalize position derivatives by Jacobian to get unit tangent vector that points
67 // in the direction of curve/beam
68 for (auto k_dim = 0U; k_dim < 3U; ++k_dim) {
69 qp_pos_deriv(j_qp, k_dim) /= jacobian;
70 }
71 }
72 }
73};

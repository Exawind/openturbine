180 YAML::Node& root, const std::string& element_type, const MapType& map
181 ) {
182 if (map.empty()) {
183 return;
184 }
185
186 YAML::Node element_node;
187
188 // Get all keys from the map and sort them to maintain consistent order
189 std::vector<size_t> keys;
190 keys.reserve(map.size());
191 for (const auto& [id, _] : map) {
192 keys.push_back(id);
193 }
194 std::sort(keys.begin(), keys.end());
195
196 // Add each element in sorted order of id
197 for (const auto& id : keys) {
198 YAML::Node array_node;
199 std::for_each(map.at(id).begin(), map.at(id).end(), [&array_node](const auto& node_id) {
200 array_node.push_back(node_id);
201 });
202
203 array_node.SetStyle(YAML::EmitterStyle::Flow);
204 element_node[std::to_string(id)] = array_node;
205 }

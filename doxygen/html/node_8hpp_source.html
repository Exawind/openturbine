9/*
10 * @brief Represents a node in the finite element model
11 *
12 * @details A node is a point in 3D space that can have position, orientation, displacement,
13 * velocity, and acceleration. Nodes are the fundamental building blocks that connect elements
14 * in the structural model. Each node has:
15 * - Unique identifier (id) -> used to reference the node in elements and constraints
16 * - Initial position and orientation (x0) -> 7 x 1 vector
17 * - Displacement from initial position (u) -> 7 x 1 vector
18 * - Velocity (v) -> 6 x 1 vector
19 * - Acceleration (vd) -> 6 x 1 vector
20 * - Parametric position within an element (s) -> scalar
21 *
22 * The position and displacement vectors contain 7 components: [x, y, z, qw, qx, qy, qz]
23 * where the first 3 are translational and the last 4 represent orientation as a quaternion.
24 *
25 * The velocity and acceleration vectors contain 6 components: [vx, vy, vz, wx, wy, wz]
26 * where the first 3 are translational and the last 3 are rotational components.
27 */
28struct Node {
29 size_t id; //< Node identifier
30 Array_7 x0; //< Initial node positions and orientations
31 Array_7 u; //< Node displacement
32 Array_6 v; //< Node velocity
33 Array_6 vd; //< Node acceleration
34 double s; //< Position of node in element on range [0, 1]
35

38 : id(node_id),
39 x0(Array_7{0., 0., 0., 1., 0., 0., 0.}),
40 u(Array_7{0., 0., 0., 1., 0., 0., 0.}),
41 v(Array_6{0., 0., 0., 0., 0., 0.}),
42 vd(Array_6{0., 0., 0., 0., 0., 0.}),
43 s(0.) {}

58 /*
59 * @brief Get the displaced position (initial position + displacement)
60 * @return Array_7 containing displaced position and orientation
61 */
62 [[nodiscard]] Array_7 DisplacedPosition() const {
63 Array_7 displaced_position{0., 0., 0., 1., 0., 0., 0.};
64
65 // Add translational components (x, y, z)
66 displaced_position[0] = this->x0[0] + this->u[0];
67 displaced_position[1] = this->x0[1] + this->u[1];
68 displaced_position[2] = this->x0[2] + this->u[2];
69
70 // Compose quaternions for orientation (w, i, j, k)
71 auto q_displaced = QuaternionCompose(
72 {this->x0[3], this->x0[4], this->x0[5], this->x0[6]}, // initial orientation
73 {this->u[3], this->u[4], this->u[5], this->u[6]} // displacement orientation
74 );
75 displaced_position[3] = q_displaced[0];
76 displaced_position[4] = q_displaced[1];
77 displaced_position[5] = q_displaced[2];
78 displaced_position[6] = q_displaced[3];
79
80 return displaced_position;
81 }

101 Node& RotateAboutPoint(const Array_4& q, const Array_3& point) {
102 //----------------------------------------------------
103 // Position, x(0:2)
104 //----------------------------------------------------
105 // Get vector from reference point to initial node position
106 const Array_3 r{this->x0[0] - point[0], this->x0[1] - point[1], this->x0[2] - point[2]};

152 //----------------------------------------------------
153 // Displacement position, u(0:2)
154 //----------------------------------------------------
155 // Get vector from reference point to displaced node position
156 const auto displaced_position = this->DisplacedPosition();
157 const Array_3 r{
158 displaced_position[0] - point[0], displaced_position[1] - point[1],
159 displaced_position[2] - point[2]

161
162 // Rotate the vector r using the provided quaternion q

165 // Update the displacement position by adding the rotated vector to the reference point

198 void SetVelocityAboutPoint(const Array_6& velocity, const Array_3& point) {
199 // Get vector from reference point to displaced node position
200 const auto displaced_position = this->DisplacedPosition();
201 const Array_3 r{
202 displaced_position[0] - point[0], displaced_position[1] - point[1],
203 displaced_position[2] - point[2]
204 };
205
206 // Calculate velocity contribution from angular velocity
207 const Array_3 omega{velocity[3], velocity[4], velocity[5]};
208 const auto omega_cross_r = CrossProduct(omega, r);
209
210 // Set node translational velocity
211 this->v[0] = velocity[0] + omega_cross_r[0];
212 this->v[1] = velocity[1] + omega_cross_r[1];
213 this->v[2] = velocity[2] + omega_cross_r[2];
214
215 // Set node angular velocity
216 this->v[3] = velocity[3];
217 this->v[4] = velocity[4];
218 this->v[5] = velocity[5];
219 }

221 //--------------------------------------------------------------------------
222 // Modify node acceleration (vd)
223 //--------------------------------------------------------------------------
224
226 void SetAccelerationAboutPoint(
227 const Array_6& acceleration, const Array_3& omega, const Array_3& point
228 ) {
229 // Get vector from reference point to displaced node position
230 const auto displaced_position = this->DisplacedPosition();
231 const Array_3 r{
232 displaced_position[0] - point[0], displaced_position[1] - point[1],
233 displaced_position[2] - point[2]
234 };
235
236 // Calculate translational acceleration contribution from angular velocity
237 const Array_3 alpha{acceleration[3], acceleration[4], acceleration[5]};
238 const auto alpha_cross_r = CrossProduct(alpha, r);
239 const auto omega_cross_omega_cross_r = CrossProduct(omega, CrossProduct(omega, r));
240
241 // Set node translational acceleration
242 this->vd[0] = acceleration[0] + alpha_cross_r[0] + omega_cross_omega_cross_r[0];
243 this->vd[1] = acceleration[1] + alpha_cross_r[1] + omega_cross_omega_cross_r[1];
244 this->vd[2] = acceleration[2] + alpha_cross_r[2] + omega_cross_omega_cross_r[2];
245
246 // Set node angular acceleration
247 this->vd[3] = alpha[0];
248 this->vd[4] = alpha[1];
249 this->vd[5] = alpha[2];
250 }

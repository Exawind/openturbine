18 const Quaternion& q, const RotationMatrix& R
19) {
20 R(0, 0) = q(0) * q(0) + q(1) * q(1) - q(2) * q(2) - q(3) * q(3);
21 R(0, 1) = 2. * (q(1) * q(2) - q(0) * q(3));

28 R(2, 2) = q(0) * q(0) - q(1) * q(1) - q(2) * q(2) + q(3) * q(3);

59inline Array_4 RotationMatrixToQuaternion(const Array_3x3& m) {
60 auto m22_p_m33 = m[1][1] + m[2][2];
61 auto m22_m_m33 = m[1][1] - m[2][2];
62 Array_4 vals{
63 m[0][0] + m22_p_m33,
64 m[0][0] - m22_p_m33,
65 -m[0][0] + m22_m_m33,
66 -m[0][0] - m22_m_m33,
67 };
68
69 // Get maximum value and index of maximum value
70 const auto* max_num = std::max_element(vals.begin(), vals.end());
71 auto max_idx = std::distance(vals.cbegin(), max_num);
72
73 auto tmp = sqrt(*max_num + 1.);
74 auto c = 0.5 / tmp;
75
76 if (max_idx == 0) {
77 return Array_4{
78 0.5 * tmp,
79 (m[2][1] - m[1][2]) * c,
80 (m[0][2] - m[2][0]) * c,
81 (m[1][0] - m[0][1]) * c,
82 };
83 }
84 if (max_idx == 1) {
85 return Array_4{
86 (m[2][1] - m[1][2]) * c,
87 0.5 * tmp,
88 (m[0][1] + m[1][0]) * c,
89 (m[0][2] + m[2][0]) * c,
90 };
91 }
92 if (max_idx == 2) {
93 return Array_4{
94 (m[0][2] - m[2][0]) * c,
95 (m[0][1] + m[1][0]) * c,
96 0.5 * tmp,
97 (m[1][2] + m[2][1]) * c,
98 };
99 }
100 if (max_idx == 3) {
101 return Array_4{
102 (m[1][0] - m[0][1]) * c,
103 (m[0][2] + m[2][0]) * c,
104 (m[1][2] + m[2][1]) * c,
105 0.5 * tmp,
106 };
107 }
108 return Array_4{1., 0., 0., 0};
109}

164template <typename QuaternionInput, typename QuaternionOutput>
165KOKKOS_INLINE_FUNCTION void QuaternionInverse(
166 const QuaternionInput& q_in, const QuaternionOutput& q_out
167) {
168 auto length =
169 Kokkos::sqrt(q_in(0) * q_in(0) + q_in(1) * q_in(1) + q_in(2) * q_in(2) + q_in(3) * q_in(3));
170
171 // Inverse of a quaternion is the conjugate divided by the length
172 q_out(0) = q_in(0) / length;
173 for (int i = 1; i < 4; ++i) {
174 q_out(i) = -q_in(i) / length;
175 }
176}

206template <typename Vector, typename Quaternion>
207KOKKOS_INLINE_FUNCTION void RotationVectorToQuaternion(
208 const Vector& phi, const Quaternion& quaternion
209) {
210 const auto angle = Kokkos::sqrt(phi(0) * phi(0) + phi(1) * phi(1) + phi(2) * phi(2));
211 const auto cos_angle = Kokkos::cos(angle / 2.);
212 const auto factor = (Kokkos::abs(angle) < 1.e-12) ? 0. : Kokkos::sin(angle / 2.) / angle;
213
214 quaternion(0) = cos_angle;
215 for (int i = 1; i < 4; ++i) {
216 quaternion(i) = phi(i - 1) * factor;
217 }

284KOKKOS_INLINE_FUNCTION
285Kokkos::Array<double, 4> NormalizeQuaternion(const Kokkos::Array<double, 4>& q) {
286 const auto length_squared = q[0] * q[0] + q[1] * q[1] + q[2] * q[2] + q[3] * q[3];
287
288 // If the length is 1, our work is done
289 if (std::abs(length_squared - 1.) < kTolerance) {
290 return q;
291 }
292
293 // If the length of the quaternion is zero, return a default unit quaternion
294 if (std::abs(length_squared) < kTolerance) {
295 return Kokkos::Array<double, 4>{1., 0., 0., 0.};
296 }
297
298 // Normalize the quaternion
299 const auto length = std::sqrt(length_squared);
300 auto normalized_quaternion = Kokkos::Array<double, 4>{};
301 for (auto k = 0U; k < 4U; ++k) {
302 normalized_quaternion[k] = q[k] / length;
303 }
304 return normalized_quaternion;
305}

310inline Array_4 TangentTwistToQuaternion(const Array_3& tangent, const double twist) {
311 const auto e1 = UnitVector(tangent);
312 Array_3 temp{0., 1., 0.};
313 if (std::abs(Dot3(e1, temp)) > 0.9) {
314 temp = {1., 0., 0.};
315 }
316 const auto a = Dot3(e1, temp);
317
318 // Construct e2 orthogonal to e1 and lying in the y-z plane
319 Array_3 e2 = UnitVector({temp[0] - e1[0] * a, temp[1] - e1[1] * a, temp[2] - e1[2] * a});
320
321 // Construct e3 as cross product
322 const auto e3 = CrossProduct(e1, e2);
323
324 auto q_tan = RotationMatrixToQuaternion({{
325 {e1[0], e2[0], e3[0]},
326 {e1[1], e2[1], e3[1]},
327 {e1[2], e2[2], e3[2]},
328 }});
329
330 const auto twist_rad = twist * M_PI / 180.;
331 auto q_twist =
332 RotationVectorToQuaternion({e1[0] * twist_rad, e1[1] * twist_rad, e1[2] * twist_rad});
333
334 return QuaternionCompose(q_twist, q_tan);
335}

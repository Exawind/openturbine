18KOKKOS_INLINE_FUNCTION void QuaternionToRotationMatrix(
19 const Quaternion& q, const RotationMatrix& R
20) {
21 R(0, 0) = q(0) * q(0) + q(1) * q(1) - q(2) * q(2) - q(3) * q(3);
22 R(0, 1) = 2. * (q(1) * q(2) - q(0) * q(3));
23 R(0, 2) = 2. * (q(1) * q(3) + q(0) * q(2));
24 R(1, 0) = 2. * (q(1) * q(2) + q(0) * q(3));
25 R(1, 1) = q(0) * q(0) - q(1) * q(1) + q(2) * q(2) - q(3) * q(3);
26 R(1, 2) = 2. * (q(2) * q(3) - q(0) * q(1));
27 R(2, 0) = 2. * (q(1) * q(3) - q(0) * q(2));
28 R(2, 1) = 2. * (q(2) * q(3) + q(0) * q(1));
29 R(2, 2) = q(0) * q(0) - q(1) * q(1) - q(2) * q(2) + q(3) * q(3);
30}

61inline std::array<double, 4> RotationMatrixToQuaternion(const std::array<std::array<double, 3>, 3>& m
62) {
63 auto m22_p_m33 = m[1][1] + m[2][2];
64 auto m22_m_m33 = m[1][1] - m[2][2];
65 std::array<double, 4> vals{
66 m[0][0] + m22_p_m33,
67 m[0][0] - m22_p_m33,
68 -m[0][0] + m22_m_m33,
69 -m[0][0] - m22_m_m33,
70 };
71
72 // Get maximum value and index of maximum value
73 const auto* max_num = std::max_element(vals.begin(), vals.end());
74 auto max_idx = std::distance(vals.cbegin(), max_num);
75
76 auto tmp = sqrt(*max_num + 1.);
77 auto c = 0.5 / tmp;
78
79 if (max_idx == 0) {
80 return std::array<double, 4>{
81 0.5 * tmp,
82 (m[2][1] - m[1][2]) * c,
83 (m[0][2] - m[2][0]) * c,
84 (m[1][0] - m[0][1]) * c,
85 };
86 }
87 if (max_idx == 1) {
88 return std::array<double, 4>{
89 (m[2][1] - m[1][2]) * c,
90 0.5 * tmp,
91 (m[0][1] + m[1][0]) * c,
92 (m[0][2] + m[2][0]) * c,
93 };
94 }
95 if (max_idx == 2) {
96 return std::array<double, 4>{
97 (m[0][2] - m[2][0]) * c,
98 (m[0][1] + m[1][0]) * c,
99 0.5 * tmp,
100 (m[1][2] + m[2][1]) * c,
101 };
102 }
103 if (max_idx == 3) {
104 return std::array<double, 4>{
105 (m[1][0] - m[0][1]) * c,
106 (m[0][2] + m[2][0]) * c,
107 (m[1][2] + m[2][1]) * c,
108 0.5 * tmp,
109 };
110 }
111 return std::array<double, 4>{1., 0., 0., 0};
112}

169template <typename QuaternionInput, typename QuaternionOutput>
170KOKKOS_INLINE_FUNCTION void QuaternionInverse(
171 const QuaternionInput& q_in, const QuaternionOutput& q_out
172) {
173 auto length =
174 Kokkos::sqrt(q_in(0) * q_in(0) + q_in(1) * q_in(1) + q_in(2) * q_in(2) + q_in(3) * q_in(3));
175
176 // Inverse of a quaternion is the conjugate divided by the length
177 q_out(0) = q_in(0) / length;
178 for (auto i = 1; i < 4; ++i) {
179 q_out(i) = -q_in(i) / length;
180 }
181}

186template <typename Quaternion1, typename Quaternion2, typename QuaternionN>
187KOKKOS_INLINE_FUNCTION void QuaternionCompose(
188 const Quaternion1& q1, const Quaternion2& q2, QuaternionN& qn
189) {
190 qn(0) = q1(0) * q2(0) - q1(1) * q2(1) - q1(2) * q2(2) - q1(3) * q2(3);
191 qn(1) = q1(0) * q2(1) + q1(1) * q2(0) + q1(2) * q2(3) - q1(3) * q2(2);
192 qn(2) = q1(0) * q2(2) - q1(1) * q2(3) + q1(2) * q2(0) + q1(3) * q2(1);
193 qn(3) = q1(0) * q2(3) + q1(1) * q2(2) - q1(2) * q2(1) + q1(3) * q2(0);
194}

213template <typename Vector, typename Quaternion>
214KOKKOS_INLINE_FUNCTION void RotationVectorToQuaternion(
215 const Vector& phi, const Quaternion& quaternion
216) {
217 const auto angle = Kokkos::sqrt(phi(0) * phi(0) + phi(1) * phi(1) + phi(2) * phi(2));
218 const auto cos_angle = Kokkos::cos(angle / 2.);
219 const auto factor = (Kokkos::abs(angle) < 1.e-12) ? 0. : Kokkos::sin(angle / 2.) / angle;
220
221 quaternion(0) = cos_angle;
222 for (auto i = 1; i < 4; ++i) {
223 quaternion(i) = phi(i - 1) * factor;
224 }
225}

269inline std::array<double, 4> RotationVectorToQuaternion(const std::array<double, 3>& phi) {
270 const auto angle = std::sqrt(phi[0] * phi[0] + phi[1] * phi[1] + phi[2] * phi[2]);
271
272 if (std::abs(angle) < 1e-12) {
273 return std::array<double, 4>{1., 0., 0., 0.};
274 }
275
276 const auto sin_angle = std::sin(angle / 2.);
277 const auto cos_angle = std::cos(angle / 2.);
278 const auto factor = sin_angle / angle;
279 return std::array<double, 4>{cos_angle, phi[0] * factor, phi[1] * factor, phi[2] * factor};

291KOKKOS_INLINE_FUNCTION
292Kokkos::Array<double, 4> NormalizeQuaternion(const Kokkos::Array<double, 4>& q) {
293 const auto length_squared = q[0] * q[0] + q[1] * q[1] + q[2] * q[2] + q[3] * q[3];
294
295 // If the length is 1, our work is done
296 if (std::abs(length_squared - 1.) < 1.e-16) {
297 return q;
298 }
299
300 // If the length of the quaternion is zero, return a default unit quaternion
301 if (std::abs(length_squared) < 1.e-16) {
302 return Kokkos::Array<double, 4>{1., 0., 0., 0.};
303 }
304
305 // Normalize the quaternion
306 const auto length = std::sqrt(length_squared);
307 auto normalized_quaternion = Kokkos::Array<double, 4>{};
308 for (auto k = 0U; k < 4; ++k) {
309 normalized_quaternion[k] = q[k] / length;
310 }
311 return normalized_quaternion;
312}

317inline std::array<double, 4> TangentTwistToQuaternion(
318 const std::array<double, 3>& tangent, const double twist
319) {
320 const auto e1 = UnitVector(tangent);
321 std::array<double, 3> temp{0., 1., 0.};
322 if (std::abs(DotProduct(e1, temp)) > 0.9) {
323 temp = {1., 0., 0.};
324 }
325 const auto a = DotProduct(e1, temp);
326
327 // Construct e2 orthogonal to e1 and lying in the y-z plane
328 std::array<double, 3> e2 =
329 UnitVector({temp[0] - e1[0] * a, temp[1] - e1[1] * a, temp[2] - e1[2] * a});
330
331 // Construct e3 as cross product
332 const auto e3 = CrossProduct(e1, e2);
333
334 auto q_tan = RotationMatrixToQuaternion({{
335 {e1[0], e2[0], e3[0]},
336 {e1[1], e2[1], e3[1]},
337 {e1[2], e2[2], e3[2]},
338 }});
339
340 const auto twist_rad = twist * std::numbers::pi / 180.;
341 auto q_twist =
342 RotationVectorToQuaternion({e1[0] * twist_rad, e1[1] * twist_rad, e1[2] * twist_rad});
343
344 return QuaternionCompose(q_twist, q_tan);
345}

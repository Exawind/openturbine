16KOKKOS_INLINE_FUNCTION void QuaternionToRotationMatrix(
17 const Quaternion& q, const RotationMatrix& R
18) {
19 R(0, 0) = q(0) * q(0) + q(1) * q(1) - q(2) * q(2) - q(3) * q(3);
20 R(0, 1) = 2. * (q(1) * q(2) - q(0) * q(3));
21 R(0, 2) = 2. * (q(1) * q(3) + q(0) * q(2));
22 R(1, 0) = 2. * (q(1) * q(2) + q(0) * q(3));

24 R(1, 2) = 2. * (q(2) * q(3) - q(0) * q(1));
25 R(2, 0) = 2. * (q(1) * q(3) - q(0) * q(2));
26 R(2, 1) = 2. * (q(2) * q(3) + q(0) * q(1));
27 R(2, 2) = q(0) * q(0) - q(1) * q(1) - q(2) * q(2) + q(3) * q(3);

59inline std::array<double, 4> RotationMatrixToQuaternion(const std::array<std::array<double, 3>, 3>& m
60) {
61 auto m22_p_m33 = m[1][1] + m[2][2];
62 auto m22_m_m33 = m[1][1] - m[2][2];
63 std::array<double, 4> vals{
64 m[0][0] + m22_p_m33,
65 m[0][0] - m22_p_m33,
66 -m[0][0] + m22_m_m33,
67 -m[0][0] - m22_m_m33,
68 };
69
70 // Get maximum value and index of maximum value
71 const auto* max_num = std::max_element(vals.begin(), vals.end());
72 auto max_idx = std::distance(vals.cbegin(), max_num);
73
74 auto tmp = sqrt(*max_num + 1.);
75 auto c = 0.5 / tmp;
76
77 if (max_idx == 0) {
78 return std::array<double, 4>{
79 0.5 * tmp,
80 (m[2][1] - m[1][2]) * c,
81 (m[0][2] - m[2][0]) * c,
82 (m[1][0] - m[0][1]) * c,
83 };
84 }
85 if (max_idx == 1) {
86 return std::array<double, 4>{
87 (m[2][1] - m[1][2]) * c,
88 0.5 * tmp,
89 (m[0][1] + m[1][0]) * c,
90 (m[0][2] + m[2][0]) * c,
91 };
92 }
93 if (max_idx == 2) {
94 return std::array<double, 4>{
95 (m[0][2] - m[2][0]) * c,
96 (m[0][1] + m[1][0]) * c,
97 0.5 * tmp,
98 (m[1][2] + m[2][1]) * c,
99 };
100 }
101 if (max_idx == 3) {
102 return std::array<double, 4>{
103 (m[1][0] - m[0][1]) * c,
104 (m[0][2] + m[2][0]) * c,
105 (m[1][2] + m[2][1]) * c,
106 0.5 * tmp,
107 };
108 }
109 return std::array<double, 4>{1., 0., 0., 0};
110}

167template <typename QuaternionInput, typename QuaternionOutput>
168KOKKOS_INLINE_FUNCTION void QuaternionInverse(
169 const QuaternionInput& q_in, const QuaternionOutput& q_out
170) {
171 auto length =
172 Kokkos::sqrt(q_in(0) * q_in(0) + q_in(1) * q_in(1) + q_in(2) * q_in(2) + q_in(3) * q_in(3));
173
174 // Inverse of a quaternion is the conjugate divided by the length
175 q_out(0) = q_in(0) / length;
176 for (auto i = 1; i < 4; ++i) {
177 q_out(i) = -q_in(i) / length;
178 }
179}

185KOKKOS_INLINE_FUNCTION void QuaternionCompose(
186 const Quaternion1& q1, const Quaternion2& q2, QuaternionN& qn
187) {
188 qn(0) = q1(0) * q2(0) - q1(1) * q2(1) - q1(2) * q2(2) - q1(3) * q2(3);
189 qn(1) = q1(0) * q2(1) + q1(1) * q2(0) + q1(2) * q2(3) - q1(3) * q2(2);
190 qn(2) = q1(0) * q2(2) - q1(1) * q2(3) + q1(2) * q2(0) + q1(3) * q2(1);
191 qn(3) = q1(0) * q2(3) + q1(1) * q2(2) - q1(2) * q2(1) + q1(3) * q2(0);

211template <typename Vector, typename Quaternion>
212KOKKOS_INLINE_FUNCTION void RotationVectorToQuaternion(
213 const Vector& phi, const Quaternion& quaternion
214) {
215 const auto angle = Kokkos::sqrt(phi(0) * phi(0) + phi(1) * phi(1) + phi(2) * phi(2));
216 const auto cos_angle = Kokkos::cos(angle / 2.);
217 const auto factor = (Kokkos::abs(angle) < 1.e-12) ? 0. : Kokkos::sin(angle / 2.) / angle;
218
219 quaternion(0) = cos_angle;
220 for (auto i = 1; i < 4; ++i) {
221 quaternion(i) = phi(i - 1) * factor;
222 }
223}

267inline std::array<double, 4> RotationVectorToQuaternion(const std::array<double, 3>& phi) {
268 const auto angle = std::sqrt(phi[0] * phi[0] + phi[1] * phi[1] + phi[2] * phi[2]);
269
270 if (std::abs(angle) < 1e-12) {
271 return std::array<double, 4>{1., 0., 0., 0.};
272 }
273
274 const auto sin_angle = std::sin(angle / 2.);
275 const auto cos_angle = std::cos(angle / 2.);
276 const auto factor = sin_angle / angle;
277 return std::array<double, 4>{cos_angle, phi[0] * factor, phi[1] * factor, phi[2] * factor};

289KOKKOS_INLINE_FUNCTION
290Kokkos::Array<double, 4> NormalizeQuaternion(const Kokkos::Array<double, 4>& q) {
291 const auto length_squared = q[0] * q[0] + q[1] * q[1] + q[2] * q[2] + q[3] * q[3];
292
293 // If the length is 1, our work is done
294 if (std::abs(length_squared - 1.) < 1.e-16) {
295 return q;
296 }
297
298 // If the length of the quaternion is zero, return a default unit quaternion
299 if (std::abs(length_squared) < 1.e-16) {
300 return Kokkos::Array<double, 4>{1., 0., 0., 0.};
301 }
302
303 // Normalize the quaternion
304 const auto length = std::sqrt(length_squared);
305 auto normalized_quaternion = Kokkos::Array<double, 4>{};
306 for (auto k = 0U; k < 4U; ++k) {
307 normalized_quaternion[k] = q[k] / length;
308 }
309 return normalized_quaternion;
310}

315inline std::array<double, 4> TangentTwistToQuaternion(
316 const std::array<double, 3>& tangent, const double twist
317) {
318 const auto e1 = UnitVector(tangent);
319 std::array<double, 3> temp{0., 1., 0.};
320 if (std::abs(DotProduct(e1, temp)) > 0.9) {
321 temp = {1., 0., 0.};
322 }
323 const auto a = DotProduct(e1, temp);
324
325 // Construct e2 orthogonal to e1 and lying in the y-z plane
326 std::array<double, 3> e2 =
327 UnitVector({temp[0] - e1[0] * a, temp[1] - e1[1] * a, temp[2] - e1[2] * a});
328
329 // Construct e3 as cross product
330 const auto e3 = CrossProduct(e1, e2);
331
332 auto q_tan = RotationMatrixToQuaternion({{
333 {e1[0], e2[0], e3[0]},
334 {e1[1], e2[1], e3[1]},
335 {e1[2], e2[2], e3[2]},
336 }});
337
338 const auto twist_rad = twist * M_PI / 180.;
339 auto q_twist =
340 RotationVectorToQuaternion({e1[0] * twist_rad, e1[1] * twist_rad, e1[2] * twist_rad});
341
342 return QuaternionCompose(q_twist, q_tan);
343}

21inline std::vector<double> MapGeometricLocations(const std::vector<double>& geom_locations) {
22 // Get first and last points of the input domain (assumed to be sorted)
23 const double domain_start = geom_locations.front();
24 const double domain_end = geom_locations.back();
25 if (domain_end == domain_start) {
26 throw std::invalid_argument(
27 "Invalid geometric locations: domain start and end points are equal."
28 );
29 }
30
31 // Map each point from domain -> [-1, 1]
32 std::vector<double> mapped_locations(geom_locations.size());
33 const auto domain_span = domain_end - domain_start;
34 for (size_t i = 0; i < geom_locations.size(); ++i) {
35 mapped_locations[i] = 2. * (geom_locations[i] - domain_start) / domain_span - 1.;
36 }
37 return mapped_locations;
38}

91 }

93 // Construct matrix A in LHS (p x p)
94 const auto A = Kokkos::View<double**, Kokkos::LayoutLeft, Kokkos::HostSpace>("A", p, p);
95 A(0, 0) = 1.;
96 A(p - 1, p - 1) = 1.;
97 for (auto j = 0U; j < p; ++j) {
98 for (auto i = 1U; i < p - 1; ++i) {
99 auto sum = 0.;
100 for (auto k = 0U; k < n; ++k) {
101 sum += shape_functions[i][k] * shape_functions[j][k];
102 }
103 A(i, j) = sum;
104 }
105 }
106
107 // Construct matrix B in RHS (p x 3)
108 const auto B = Kokkos::View<double* [3], Kokkos::LayoutLeft, Kokkos::HostSpace>("B", p);
109 for (auto dim = 0U; dim < 3U; ++dim) {
110 B(0, dim) = points_to_fit[0][dim];
111 B(p - 1, dim) = points_to_fit[n - 1][dim];
112 }
113 for (auto i = 1U; i < p - 1; ++i) {
114 for (auto k = 0U; k < n; ++k) {
115 for (auto dim = 0U; dim < 3U; ++dim) {
116 B(i, dim) += shape_functions[i][k] * points_to_fit[k][dim];
117 }
118 }
119 }
120
121 // Solve the least squares problem for each dimension of B
122 const auto IPIV = Kokkos::View<int*, Kokkos::LayoutLeft, Kokkos::HostSpace>("IPIV", B.extent(0));
123
124 KokkosLapack::gesv(A, B, IPIV);

131 }
132 }

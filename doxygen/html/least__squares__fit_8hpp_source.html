23inline std::vector<double> MapGeometricLocations(const std::vector<double>& geom_locations) {
24 // Get first and last points of the input domain (assumed to be sorted)
25 const double domain_start = geom_locations.front();
26 const double domain_end = geom_locations.back();
27 if (domain_end == domain_start) {
28 throw std::invalid_argument(
29 "Invalid geometric locations: domain start and end points are equal."
30 );
31 }
32
33 // Map each point from domain -> [-1, 1]
34 std::vector<double> mapped_locations(geom_locations.size());
35 const auto domain_span = domain_end - domain_start;
36 for (size_t i = 0; i < geom_locations.size(); ++i) {
37 mapped_locations[i] = 2. * (geom_locations[i] - domain_start) / domain_span - 1.;
38 }
39 return mapped_locations;
40}

92 }
93 const size_t n = shape_functions[0].size();
94 for (const auto& row : shape_functions) {
95 if (row.size() != n) {
96 throw std::invalid_argument("Inconsistent number of columns in shape_functions.");
97 }
98 }
99
100 // Construct matrix A in LHS (p x p)
101 const auto A = Kokkos::View<double**, Kokkos::LayoutLeft, Kokkos::HostSpace>("A", p, p);
102 A(0, 0) = 1.;
103 A(p - 1, p - 1) = 1.;
104 for (auto j = 0U; j < p; ++j) {
105 for (auto i = 1U; i < p - 1; ++i) {
106 auto sum = 0.;
107 for (auto k = 0U; k < n; ++k) {
108 sum += shape_functions[i][k] * shape_functions[j][k];
109 }
110 A(i, j) = sum;
111 }
112 }
113
114 // Construct matrix B in RHS (p x 3)
115 const auto B = Kokkos::View<double* [3], Kokkos::LayoutLeft, Kokkos::HostSpace>("B", p);
116 for (auto dim = 0U; dim < 3U; ++dim) {
117 B(0, dim) = points_to_fit[0][dim];
118 B(p - 1, dim) = points_to_fit[n - 1][dim];
119 }
120 for (auto i = 1U; i < p - 1; ++i) {
121 for (auto k = 0U; k < n; ++k) {
122 for (auto dim = 0U; dim < 3U; ++dim) {
123 B(i, dim) += shape_functions[i][k] * points_to_fit[k][dim];
124 }
125 }
126 }
127
128 // Solve the least squares problem for each dimension of B
129 const auto IPIV = Kokkos::View<int*, Kokkos::LayoutLeft, Kokkos::HostSpace>("IPIV", B.extent(0));

131 KokkosLapack::gesv(A, B, IPIV);
132
133 auto result = std::vector<std::array<double, 3>>(B.extent(0));
134

136 for (auto j = 0U; j < result.front().size(); ++j) {

138 }
139 }

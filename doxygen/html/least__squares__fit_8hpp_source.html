23inline std::vector<double> MapGeometricLocations(const std::vector<double>& geom_locations) {
24 // Get first and last points of the input domain (assumed to be sorted)
25 const double domain_start = geom_locations.front();
26 const double domain_end = geom_locations.back();
27 if (domain_end == domain_start) {
28 throw std::invalid_argument(
29 "Invalid geometric locations: domain start and end points are equal."
30 );
31 }
32
33 // Map each point from domain -> [-1, 1]
34 std::vector<double> mapped_locations(geom_locations.size());
35 const auto domain_span = domain_end - domain_start;
36 for (size_t i = 0; i < geom_locations.size(); ++i) {
37 mapped_locations[i] = 2. * (geom_locations[i] - domain_start) / domain_span - 1.;
38 }
39 return mapped_locations;
40}

125inline std::vector<std::array<double, 3>> PerformLeastSquaresFitting(
126 size_t p, const Matrix& shape_functions, const std::vector<std::array<double, 3>>& points_to_fit
127) {
128 if (shape_functions.size() != p) {
129 throw std::invalid_argument("shape_functions rows do not match order p.");
130 }
131 const size_t n = shape_functions[0].size();
132 if (std::any_of(shape_functions.begin(), shape_functions.end(), [n](const auto& row) {
133 return row.size() != n;
134 })) {
135 throw std::invalid_argument("Inconsistent number of columns in shape_functions.");
136 }
137
138 // Construct matrix A in LHS (p x p)
139 const auto A = Kokkos::View<double**, Kokkos::LayoutLeft, Kokkos::HostSpace>("A", p, p);
140 A(0, 0) = 1.;
141 A(p - 1, p - 1) = 1.;
142 for (auto j = 0U; j < p; ++j) {
143 for (auto i = 1U; i < p - 1; ++i) {
144 auto sum = 0.;
145 for (auto k = 0U; k < n; ++k) {
146 sum += shape_functions[i][k] * shape_functions[j][k];
147 }
148 A(i, j) = sum;
149 }
150 }
151
152 // Construct matrix B in RHS (p x 3)
153 const auto B = Kokkos::View<double* [3], Kokkos::LayoutLeft, Kokkos::HostSpace>("B", p);
154 for (auto dim = 0U; dim < 3U; ++dim) {
155 B(0, dim) = points_to_fit[0][dim];
156 B(p - 1, dim) = points_to_fit[n - 1][dim];
157 }
158 for (auto i = 1U; i < p - 1; ++i) {
159 for (auto k = 0U; k < n; ++k) {
160 for (auto dim = 0U; dim < 3U; ++dim) {
161 B(i, dim) += shape_functions[i][k] * points_to_fit[k][dim];
162 }
163 }
164 }
165
166 // Solve the least squares problem for each dimension of B
167 const auto IPIV = Kokkos::View<int*, Kokkos::LayoutLeft, Kokkos::HostSpace>("IPIV", B.extent(0));
168
169 KokkosLapack::gesv(A, B, IPIV);
170
171 auto result = std::vector<std::array<double, 3>>(B.extent(0));
172
173 for (auto i = 0U; i < result.size(); ++i) {
174 for (auto j = 0U; j < result.front().size(); ++j) {
175 result[i][j] = B(i, j);
176 }
177 }
178
179 return result;
180}

151 void CreateNodeGeometry(const BeamInput& input) {
152 const auto n_nodes = input.element_order + 1;
153 const auto n_geometry_pts =
154 std::min({input.ref_axis.coordinate_grid.size(), n_nodes, kMaxGeometryPoints});
155
156 if (n_geometry_pts < n_nodes) {
157 // We need to project from n_geometry_pts -> element_order
158 const std::vector<double> kp_xi(MapGeometricLocations(input.ref_axis.coordinate_grid));
159 const auto [phi_kn_geometry, phi_prime_kn_geometry] =
160 ShapeFunctionMatrices(kp_xi, GenerateGLLPoints(n_geometry_pts - 1));
161 const auto geometry_points = PerformLeastSquaresFitting(
162 n_geometry_pts, phi_kn_geometry, input.ref_axis.coordinates
163 );
164 const auto node_coords =
165 ProjectPointsToTargetPolynomial(n_geometry_pts, n_nodes, geometry_points);
166
167 this->node_coordinates.clear();
168 this->node_coordinates.reserve(node_coords.size());
169 std::copy(
170 node_coords.begin(), node_coords.end(), std::back_inserter(this->node_coordinates)
171 );
172 } else {
173 // Fit node coordinates to key points
174 const std::vector<double> kp_xi(MapGeometricLocations(input.ref_axis.coordinate_grid));
175 const auto [phi_kn, phi_prime_kn] = ShapeFunctionMatrices(kp_xi, this->node_xi);
176 this->node_coordinates =
177 PerformLeastSquaresFitting(n_nodes, phi_kn, input.ref_axis.coordinates);
178 }
179
180 // Calculate tangent vectors at each node
181 this->CalcNodeTangents();
182 }

190 // Add nodes to model
191 std::vector<size_t> node_ids;
192 for (auto i = 0U; i < this->node_xi.size(); ++i) {
193 const auto& pos = this->node_coordinates[i];
194 const auto q_rot = TangentTwistToQuaternion(this->node_tangents[i], 0.);
195 const auto node_id =
196 model.AddNode()
197 .SetElemLocation((this->node_xi[i] + 1.) / 2.)
198 .SetPosition(pos[0], pos[1], pos[2], q_rot[0], q_rot[1], q_rot[2], q_rot[3])
199 .Build();

226 void PositionBladeInSpace(const BeamInput& input, Model& model) const {
227 // Extract root location, orientation, and velocity
228 const std::array<double, 3> root_location{
229 input.root.position[0], input.root.position[1], input.root.position[2]

232 input.root.position[3], input.root.position[4], input.root.position[5],

249 model.SetBeamAccelerationAboutPoint(

269 void CalcNodeTangents() {
270 const auto n_nodes{this->node_coordinates.size()};
271
272 // Calculate the derivative shape function matrix for the nodes
273 const auto [phi, phi_prime] = ShapeFunctionMatrices(this->node_xi, this->node_xi);
274
275 // Calculate tangent vectors for each node
276 this->node_tangents.resize(n_nodes, {0., 0., 0.});
277 for (auto i = 0U; i < n_nodes; ++i) {
278 for (auto j = 0U; j < 3; ++j) {
279 for (auto k = 0U; k < n_nodes; ++k) {
280 this->node_tangents[i][j] += phi_prime[k][i] * this->node_coordinates[k][j];
281 }
282 }
283 }
284
285 // Normalize tangent vectors
286 std::transform(
287 this->node_tangents.begin(), this->node_tangents.end(), this->node_tangents.begin(),
288 [](std::array<double, 3>& tangent) {
289 const auto norm = Norm(tangent);
290 std::transform(tangent.begin(), tangent.end(), tangent.begin(), [norm](double v) {
291 return v / norm;
292 });
293 return tangent;
294 }
295 );
296 }

303 static std::vector<BeamSection> BuildBeamSections(const BeamInput& input) {
304 // Extraction section stiffness and mass matrices from blade definition
305 std::vector<BeamSection> sections;
306
307 // Add first section after rotating matrices to account for twist
308 auto twist = LinearInterp(
309 input.sections[0].location, input.ref_axis.twist_grid, input.ref_axis.twist
310 );
311 auto q_twist = RotationVectorToQuaternion({twist * M_PI / 180., 0., 0.});
312 sections.emplace_back(
313 input.sections[0].location, RotateMatrix6(input.sections[0].mass_matrix, q_twist),
314 RotateMatrix6(input.sections[0].stiffness_matrix, q_twist)
315 );
316
317 // Loop through remaining section locations
318 for (auto i = 1U; i < input.sections.size(); ++i) {
319 // Add refinement sections if requested
320 for (auto j = 0U; j < input.section_refinement; ++j) {
321 // Calculate interpolation ratio between bounding sections
322 const auto alpha =
323 static_cast<double>(j + 1) / static_cast<double>(input.section_refinement + 1);
324
325 // Interpolate grid location
326 const auto grid_value = (1. - alpha) * input.sections[i - 1].location +
327 alpha * input.sections[i].location;
328
329 // Interpolate mass and stiffness matrices from bounding sections
330 Array_6x6 mass_matrix;
331 Array_6x6 stiffness_matrix;
332 for (auto mi = 0U; mi < 6; ++mi) {
333 for (auto ni = 0U; ni < 6; ++ni) {
334 mass_matrix[mi][ni] =
335 (1. - alpha) * input.sections[i - 1].mass_matrix[mi][ni] +
336 alpha * input.sections[i].mass_matrix[mi][ni];
337 stiffness_matrix[mi][ni] =
338 (1. - alpha) * input.sections[i - 1].stiffness_matrix[mi][ni] +
339 alpha * input.sections[i].stiffness_matrix[mi][ni];
340 }
341 }
342
343 // Calculate twist at current section location via linear interpolation
344 twist = LinearInterp(
345 input.sections[i].location, input.ref_axis.twist_grid, input.ref_axis.twist
346 );
347
348 // Add refinement section
349 q_twist = RotationVectorToQuaternion({twist * M_PI / 180., 0., 0.});
350 sections.emplace_back(
351 grid_value, RotateMatrix6(mass_matrix, q_twist),
352 RotateMatrix6(stiffness_matrix, q_twist)
353 );
354 }
355
356 // Add ending section
357 twist = LinearInterp(
358 input.sections[i].location, input.ref_axis.twist_grid, input.ref_axis.twist
359 );
360 q_twist = RotationVectorToQuaternion({twist * M_PI / 180., 0., 0.});
361 sections.emplace_back(
362 input.sections[i].location, RotateMatrix6(input.sections[i].mass_matrix, q_twist),
363 RotateMatrix6(input.sections[i].stiffness_matrix, q_twist)
364 );
365 }
366
367 return sections;
368 }
